package org.jsxm.maven.plugin.jsxmtool.animation.batchanimation;

import java.util.LinkedList;
import java.util.List;
import java.util.concurrent.ExecutorService;

import org.jsxm.maven.plugin.jsxmtool.core.JSXM;
import org.jsxm.maven.plugin.jsxmtool.core.JSXMThreadPoolFactory;
import org.jsxm.maven.plugin.jsxmtool.core.Specification;
import org.jsxm.maven.plugin.jsxmtool.runtimeexceptions.InputXMLNotFoundException;
import org.jsxm.maven.plugin.jsxmtool.utilities.SxmCompilationUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.slf4j.MDC;

/**
 * @author Konstantinos Margaritis
 * 
 */
public class SxmBanimationManager {
    
    private final Logger logger = (Logger) LoggerFactory.getLogger(this.getClass());
    private boolean banimateAll;
    private List<Specification> specificationList = new LinkedList<Specification>();
    private List<String> banimateList = new LinkedList<String>();
    private boolean silent;
    
    /**
     * Instantiates a new sxm banimation manager.
     *
     * @param banimateAll the banimate all
     * @param banimateList the banimate list
     * @param silent the silent
     */
    public SxmBanimationManager(boolean banimateAll, List<String> banimateList, boolean silent) {
        this.banimateAll = banimateAll;
        this.banimateList = banimateList;
        this.silent = silent;
    }
    
    /**
     * Banimate.
     */
    public void banimate() {
        long banimatePhaseStart = System.currentTimeMillis();
        if (!banimateAll) {
            logger.debug("" + banimateAll);
            createBatchAnimationList();
            threadPoolCreator(specificationList);
        } else {
            logger.debug("List size " + JSXM.getInstance().getSpecificationList().size());
            for (Specification spec : JSXM.getInstance().getSpecificationList()) {
                logger.debug(spec.getClassName());
            }
            threadPoolCreator(JSXM.getInstance().getSpecificationList());
        }
        long banimatePhaseEnd = System.currentTimeMillis();
        long banimatePhase=banimatePhaseEnd-banimatePhaseStart;
        logger.info("---  Banimating phase  : ..."+banimatePhase +" ms\n");
        if (!JSXM.getInstance().getMissingInputs().isEmpty()) {
            logger.debug("Missing inputs size: " + JSXM.getInstance().getMissingInputs().size());
            logger.debug("Missing inputs list: " + JSXM.getInstance().getMissingInputs());
            logger.info("Resolve following error:");
            for (int i = 0; i < JSXM.getInstance().getMissingInputs().size(); i++) {
                MDC.put("specificationName", JSXM.getInstance().getMissingInputs().get(i));
                logger.error("           -->Input (" + JSXM.getInstance().getMissingInputs().get(i)
                        + "_input.xml) file missing for batch animation.<--");
                MDC.remove("specificationName");
            }
            throw new InputXMLNotFoundException("\nInput (_input.xml) file missing for batch animation.<--");
        }
    }
    
    /**
     * Creates the batch animation list.
     */
    private void createBatchAnimationList() {
        for (int i = 0; i < banimateList.size(); i++) {
            for (int j = 0; j < JSXM.getInstance().getSpecificationList().size(); j++) {
                logger.debug(banimateList.get(i));
                logger.debug(SxmCompilationUtils.extractSourceClass(JSXM.getInstance().getSpecificationList().get(j)
                        .getName(), JSXM.SPECIFICATION_EXTENSION_LENGTH));
                if (banimateList.get(i).equals(
                        SxmCompilationUtils.extractSourceClass(JSXM.getInstance().getSpecificationList().get(j)
                                .getName(), JSXM.SPECIFICATION_EXTENSION_LENGTH))) {
                    specificationList.add(JSXM.getInstance().getSpecificationList().get(j));
                    logger.debug("Match found : Banimating...");
                }
            }
        }
    }
    
    /**
     * Thread pool creator.
     * 
     * @param specList
     *            the spec list
     */
    private void threadPoolCreator(List<Specification> specList) {
        ExecutorService banimateJsxmExecutor=JSXMThreadPoolFactory.getInstance().getExecutor(JSXMThreadPoolFactory.getInstance().getBanimationThreadPool());
        for (int j = 0; j < specList.size(); j++) {
            Specification p = specList.get(j);
            Runnable banimateJsxmWorker = new SxmBanimationThread(silent, p);
            banimateJsxmExecutor.execute(banimateJsxmWorker);
        }
        /*
         * This will make the executor accept no new threads and finish all
         * existing threads in queue
         */
        banimateJsxmExecutor.shutdown();
        /*
         * Waiting for threads to finish execution
         */
        while (!banimateJsxmExecutor.isTerminated()) {
            logger.trace("Processing banimation threads");
        }
    }
    
}
