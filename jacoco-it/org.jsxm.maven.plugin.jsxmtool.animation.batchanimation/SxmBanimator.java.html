package org.jsxm.maven.plugin.jsxmtool.animation.batchanimation;

import java.io.File;
import java.util.List;
import org.jdom.Attribute;
import org.jdom.Document;
import org.jdom.Element;
import org.jdom.Namespace;
import org.jsxm.jsxmcore.core.AbstractDefsxmFactory;
import org.jsxm.jsxmcore.core.Input;
import org.jsxm.jsxmcore.core.Output;
import org.jsxm.jsxmcore.core.SXM;
import org.jsxm.jsxmcore.exceptions.DefinitionNotFoundException;
import org.jsxm.jsxmcore.exceptions.InputNotFound;
import org.jsxm.jsxmcore.exceptions.NullSXMValueException;
import org.jsxm.jsxmcore.runtimeexceptions.CallElementException;
import org.jsxm.jsxmcore.util.Various;
import org.jsxm.jsxmcore.util.XML;
import org.jsxm.maven.plugin.jsxmtool.core.JSXM;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * @author Konstantinos Margaritis
 * 
 */
public class SxmBanimator {
    
    private static final Object BANIMATE_LOCK = new Object();
    private final Logger logger = (Logger) LoggerFactory.getLogger(this.getClass());
    private String outfilename = null;
    private SXM fsm;
    
    /**
     * Banimate sxm.
     * 
     * @param className
     *            the class name
     * @param silent
     *            the silent
     * @param mode
     *            the mode
     * @param animationDirectory
     *            the animation directory
     * @param tempDirectory
     *            the temp directory
     * @param specificationDirectory
     *            the specification directory
     */
    public void banimateSXM(String className, boolean silent, final String mode, String animationDirectory,
            String tempDirectory, String specificationDirectory) {
        final String specName = "org.jsxm.jsxmcore.types." + className + "SXM";
        if (mode.equals("verbose")) {
            logger.info("verbose mode on");
        }
        final String filename = animationDirectory + className + "_input.xml";
        final File input = new File(filename);
        if (input.exists()) {
            synchronized (BANIMATE_LOCK) {
                AbstractDefsxmFactory.setDeffilePath(specificationDirectory);
                logger.info("           " + "###Batch animation for " + className + "###");
                logger.info("           " + "-->Input file for animation : " + className + "_input.xml");
                logger.info("           " + "-->Output file for animation: " + className + "_output.xml");
                outfilename = animationDirectory + className + "_output.xml"; 
                fsm = null;
                try {
                    fsm = (SXM) Various.getInstance().loadClass(specName).newInstance();
                    logger.debug(fsm.toString());
                } catch (InstantiationException e) {
                    logger.error("InstantiationException : ", e);
                } catch (IllegalAccessException e) {
                    logger.error("IllegalAccessException : ", e);
                }
                setFsm(fsm);
                SXM.setSilent(silent);
                batchAnimation(filename, tempDirectory);
                if (outfilename != null && mode.equals("verbose")) {
                    logger.info("\nInputs/Outputs saved in file: " + outfilename);
                }            
                logger.info("           ###Batch animation (" + className + ") was completed succesfully.###\n ");
            }
        } else {
            synchronized (BANIMATE_LOCK) {
                JSXM.getInstance().addMissingInput(className);
            }
        }
    }
    
    /**
     * Batch animation.
     * 
     * @param filename
     *            the filename
     * @param tempDirectory
     *            the temp directory
     */
    private void batchAnimation(String filename, String tempDirectory) {
        fsm.reset();
        Document doc = XML.getDocXML(filename, "Error reading inputs file", true);
        /*
         * New sequence to place the calls
         */
        Element root = doc.getRootElement();
        Namespace ns = Namespace.getNamespace("xsi", "http://www.w3.org/2001/XMLSchema-instance");
        root.addNamespaceDeclaration(ns);
        root.setAttribute(new Attribute("schemaLocation",
                "http://www.jsxm.org/schema http://www.jsxm.org/schema/output.xsd", ns));
        
        @SuppressWarnings("unchecked")
        List<Element> calls = root.getChildren("call", Various.getInstance().getNameSpace());
        
        if (calls.size() == 0) {
            logger.error("No <call> elements in the XML file.");
            throw new CallElementException("CallElementException: No <call> elements in the XML file.");
        }
        for (Element callElement : calls) {
            Input input = null;
            try {
                try {
                    final String tempPath = tempDirectory;
                    XML x = new XML();
                    x.setPathForGenClass(tempPath);
                    logger.debug("Fsm: {}",fsm);
                    logger.debug("CallElement: {}",callElement);
                    input = XML.readInput(fsm, callElement);
                    logger.debug("Input Element: {}",input);
                } catch (NullSXMValueException e) {
                    logger.error("NullSXMValueException : ", e);
                }
            } catch (DefinitionNotFoundException e) {
                logger.error("DefinitionNotFoundException : ", e);
            } catch (InputNotFound e) {
                logger.error("InputNotFound : ", e);
            }
            Output response = null;
            try {
                response = fsm.animateStep(input, SXM.CONTINUE_ON_ERROR);
            } catch (Exception e) {
                logger.error("Exception : ", e);
                break;
            }
            
            if (response == null) {
                logger.error("Null response during animation!");
            }
            Element outputElement = XML.createOutputElement(response);
            callElement.addContent(outputElement);
        }
        if (outfilename != null) {
            XML.saveToFile(doc, outfilename);
        }
    }
    
    /**
     * @return
     */
    public SXM getFsm() {
        return fsm;
    }
    
    /**
     * @param fsm
     */
    public void setFsm(SXM fsm) {
        this.fsm = fsm;
    }
    
}
