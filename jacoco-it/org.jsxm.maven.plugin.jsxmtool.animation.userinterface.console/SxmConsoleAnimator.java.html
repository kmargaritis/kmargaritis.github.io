package org.jsxm.maven.plugin.jsxmtool.animation.userinterface.console;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.UnsupportedEncodingException;

import org.jdom.Attribute;
import org.jdom.Document;
import org.jdom.Element;
import org.jdom.Namespace;
import org.jdom.output.XMLOutputter;
import org.jsxm.jsxmcore.util.Various;
import org.jsxm.maven.plugin.jsxmtool.animation.core.SxmAbstractAnimator;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.slf4j.MDC;

/**
 * @author Konstantinos Margaritis
 * 
 */
public class SxmConsoleAnimator extends SxmAbstractAnimator {
    
    private final Logger logger = (Logger) LoggerFactory.getLogger(this.getClass());
    private static final String RESET_COMMAND = ".r.";
    private static final String INPUT_LIST_COMMAND = ".i.";
    private static final String QUIT_COMMAND = ".q.";
    private static final String HELP_COMMAND = ".h.";
    private static final String SILENT_COMMAND = ".s.";
    private static final String SAVE_COMMAND = ".log.";
    private static final int PRINTER_CHARACTER_LENGHT = 60;
    private BufferedReader bufferedReader =null;
    private boolean silent=false;
    
    
    @Override
    public void interactiveAnimation(String className, String animationDirectory, String specificationDirectory)
            throws UnsupportedEncodingException {
        MDC.put("specificationName", className);
        instantiateSXM(className, animationDirectory, specificationDirectory);
        setBufferedReader(animationInputSource("", "console",animationDirectory)) ;
        Element rootElement = new Element("sequence", Various.getInstance().getNameSpace());
        Namespace ns = Namespace.getNamespace("xsi", "http://www.w3.org/2001/XMLSchema-instance");
        rootElement.addNamespaceDeclaration(ns);
        rootElement.setAttribute(new Attribute("schemaLocation",
                "http://www.jsxm.org/schema http://www.jsxm.org/schema/input.xsd", ns));
        XMLOutputter xmlOutput = new XMLOutputter();
        Document doc = new Document(rootElement);
        logger.info("SXM Animator in interactive mode.\n");
        logger.info("Enter .h. to see a list of commands.\n");
        reset();
        animationOptions(rootElement, doc, className, animationDirectory, xmlOutput);
        MDC.remove("specificationName");
    }
    
    @Override
    public void animationOptions(Element rootElement, Document doc, String className,
            String animationDirectory, XMLOutputter xmlOutput) {
        String str = "";
        boolean running = true;
        while (running) {
            logger.info("STATE  : " + getFsm().getCurrentState());
            logger.info("MEMORY : " + getFsm().memoryString());
            printSeparator("-");
            logger.info("Input  : ");
            try {
                if (bufferedReader!= null) {
                    str = bufferedReader.readLine();
                }
            } catch (IOException e) {
                logger.error("IOException : ", e);
            }
            if (str != null) {
                if (str.equals("")) {
                    continue;
                }
                if (str.equals(RESET_COMMAND)) {
                    reset();
                    rootElement.detach();
                    rootElement = new Element("sequence", Various.getInstance().getNameSpace());
                    doc = new Document(rootElement);
                    logger.debug("New root for document {}"+doc.getRootElement());
                    continue;
                }
                if (str.equals(INPUT_LIST_COMMAND)) {
                    alphabetPrint();
                    continue;
                }
                if (str.equals(HELP_COMMAND)) {
                    help();
                    continue;
                }
                if (str.equals(SILENT_COMMAND)) {
                    getFsm().toggleSilent();
                    continue;
                }
                if (str.equals(QUIT_COMMAND)) {
                    running = false;
                }
                if (str.equals(SAVE_COMMAND)) {
                    if(!doc.getRootElement().getChildren().isEmpty()){
                        save(doc, xmlOutput, className, animationDirectory);
                        logger.info(className+" animation has been saved");
                    }else{
                        printError("NO SXM calls to save.");
                        printError("Either the machine has been reseted or the was no animation");
                    }
                    continue;
                }
                if (prepareStep(str, rootElement)) {
                    continue;
                }
            } else {
                running = false;
                logger.debug("System.in encountered error on reading. The InputStream is null");
            }
            
        }
    }
    
    @Override
    protected void alphabetPrint() {
        printSeparator("=");
        logger.info("List of inputs:");
        for (String signature : getInputs()) {
            logger.info("\t" + signature);
        }
        printSeparator("=");
    }
      
    @Override
    protected void help() {
        printSeparator("*");
        logger.info("\t" + RESET_COMMAND + "  resets the SXM to its initial state and memory.");
        logger.info("\t" + INPUT_LIST_COMMAND + "  lists available inputs.");
        logger.info("\t" + QUIT_COMMAND + "  quits the animator.");
        logger.info("\t" + HELP_COMMAND + "  displays this list.");
        logger.info("\t" + SILENT_COMMAND + "  (toggle) show less/more details in the animation.");
        logger.info("\t" + SAVE_COMMAND + "  Saves the animation.");
        printSeparator("*");
    }
        
    @Override
    protected void printSeparator(String c) {
        StringBuffer ch = new StringBuffer();
        for (int i = 0; i < PRINTER_CHARACTER_LENGHT; i++) {
            ch.append(c);
        }
        logger.info(ch.toString());
    }
    
    @Override
    protected void printInfo(String message) {
        logger.info("{}",message);
    }
    
    /**
     * Gets the buffered reader.
     *
     * @return the buffered reader
     */
    public BufferedReader getBufferedReader() {
        return bufferedReader;
    }

    /**
     * Sets the buffered reader.
     *
     * @param bufferedReader the new buffered reader
     */
    public void setBufferedReader(BufferedReader bufferedReader) {
        this.bufferedReader = bufferedReader;
    }

    /**
     * @return the silent
     */
    public boolean isSilent() {
        return silent;
    }

    /**
     * @param silent the silent to set
     */
    public void setSilent(boolean silent) {
        this.silent = silent;
    }
    
    @Override
    protected void printError(String message) {
        logger.error("*** ERROR:");
        logger.error("*** {}",message);
        logger.error("*** ");
    }


    @Override
    protected void printError(String message, Object obj) {
        logger.error("*** ERROR:");
        logger.error("*** {} \n     {}",message,obj);
        logger.error("*** ");   
    }
                       
}
