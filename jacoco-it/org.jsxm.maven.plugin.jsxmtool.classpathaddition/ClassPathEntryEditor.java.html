package org.jsxm.maven.plugin.jsxmtool.classpathaddition;

import java.io.File;
import java.util.LinkedList;
import java.util.List;

import org.jdom.Document;
import org.jdom.Element;
import org.jdom.output.Format;
import org.jdom.output.XMLOutputter;
import org.jsxm.jsxmcore.util.XML;
import org.jsxm.maven.plugin.jsxmtool.utilities.SxmCompilationUtils;
import org.jsxm.maven.plugin.jsxmtool.utilities.SxmFileUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * @author Konstantinos Margaritis
 * 
 */
public class ClassPathEntryEditor {
    
    private final Logger logger = (Logger) LoggerFactory.getLogger(this.getClass());
    private String projectName = "";
    
    /**
     * Update class path.
     * 
     * @param baseDir
     *            the base dir
     * @param jsxmSpecRootDir
     *            the jsxm spec root dir
     * @param exclusion
     *            the exclusion
     */
    public void updateClassPath(String baseDir, String jsxmSpecRootDir, String exclusion,boolean topView) {
        File classPath = new File(baseDir + "/.classpath");
        logger.debug(classPath.getAbsolutePath());
        logger.debug("" + classPath.exists());
        
        if(topView){
            File specDirectory = new File(jsxmSpecRootDir);
            logger.debug(specDirectory.getAbsolutePath());
            logger.debug("" + specDirectory.exists());
            
            List<File> specificationFolderList = new LinkedList<File>();
            List<File> packagesList = new LinkedList<File>();
            
            SxmFileUtils.traverseDirectoryWithExclusion(specDirectory, packagesList, ".xml");
            logger.debug("" + specificationFolderList);
            
            SxmCompilationUtils.createSpecNameFolderList(packagesList, specificationFolderList);
            logger.debug("" + packagesList);
            
            List<String> listClassPathEntries = new LinkedList<String>();
            Document doc = XML.getDocXML(classPath.getAbsolutePath(), "Problem occured", true);
            Element rootElement = doc.getRootElement();
            
            createSpecificationBuildPathNonStructured(specificationFolderList, rootElement, listClassPathEntries, baseDir);
            addClassPathEntries(cleanSpecificationList(specificationFolderList, listClassPathEntries, baseDir),
                    rootElement, exclusion);
            saveXML(doc, classPath);
        }else{
            Document doc = XML.getDocXML(classPath.getAbsolutePath(), "Problem occured", true);
            Element rootElement = doc.getRootElement();
            addUniqueClassPathEntry(rootElement, "**/animation/|**/specification/|**/testAdapters/|**/tests/|**/xjcgen/|jsxmLogs/","src/spec");
            saveXML(doc, classPath);
        }
    }
    
    /**
     * Clean specification list.
     * 
     * @param specificationFolderList
     *            the specification folder list
     * @param listClassPathEntries
     *            the list class path entries
     * @param baseDir
     *            the base dir
     * @return the list
     */
    public List<String> cleanSpecificationList(List<File> specificationFolderList, List<String> listClassPathEntries,
            String baseDir) {
        List<String> stringSpecList = new LinkedList<String>();
        for (int i = 0; i < specificationFolderList.size(); i++) {
            stringSpecList.add(new File(specificationFolderList.get(i).getParent()).getParent().replace(baseDir + "/",
                    ""));
        }
        logger.debug("" + listClassPathEntries);
        logger.debug("" + stringSpecList);
        stringSpecList.removeAll(listClassPathEntries);
        return stringSpecList;
    }
    
    /**
     * Creates the specification build path non structured.
     * 
     * @param specificationFolderList
     *            the specification folder list
     * @param root
     *            the root
     * @param listClassPathEntries
     *            the list class path entries
     * @param baseDir
     *            the base dir
     */
    public void createSpecificationBuildPathNonStructured(List<File> specificationFolderList, Element root,
            List<String> listClassPathEntries, String baseDir) {
        for (File file : specificationFolderList) {
            /*
             * specification folder
             */
            File parent = new File(file.getParent());
            /*
             * parent of specification
             */
            File grandParent = new File(parent.getParent());
            String path = grandParent.getAbsolutePath().replace(baseDir + "/", "");
            logger.debug(path);
            String pathAttribute="path";
            for (int i = 0; i < root.getChildren().size(); i++) {
                Element classpathentry = (Element) root.getChildren().get(i);
                logger.debug("" + classpathentry);
                classpathentry.getAttribute(pathAttribute);
                logger.debug("Attribute " + classpathentry.getAttribute(pathAttribute).getValue());
                logger.debug("Path " + path);
                if (classpathentry.getAttribute(pathAttribute).getValue().equals(path)) {
                    logger.debug("classPathEntry already exists.");
                    logger.debug("" + classpathentry);
                    listClassPathEntries.add(path);
                }
            }
        }
    }
    
    /**
     * Adds the class path entries.
     * 
     * @param specsToAdd
     *            the specs to add
     * @param root
     *            the root
     * @param exclusion
     *            the exclusion
     */
    public void addClassPathEntries(List<String> specsToAdd, Element root, String exclusion) {
        for (int i = 0; i < specsToAdd.size(); i++) {
            addClassPathEntry(root, exclusion,  specsToAdd.get(i));
        }
    }
    
    
    /**
     * Adds the unique class path entry.
     *
     * @param rootElement the root element
     * @param exclusion the exclusion
     * @param path the path
     * @return true, if successful
     */
    public boolean addUniqueClassPathEntry(Element rootElement,String exclusion,String path){
        boolean exists=false;
        String excluding="excluding";
        for (int i = 0; i < rootElement.getChildren().size(); i++) {
            Element classpathentry = (Element) rootElement.getChildren().get(i);
            logger.debug("" + classpathentry);
            if(classpathentry.getAttribute(excluding)!=null){
                classpathentry.getAttribute(excluding);
                logger.debug("Attribute " + classpathentry.getAttribute(excluding).getValue());
                logger.debug("Path " + path);
                if (classpathentry.getAttribute(excluding).getValue().equals(exclusion)) {
                    logger.debug("classParhEntry already exists.");
                    logger.debug("" + classpathentry);
                    exists= true;
                }
            }
        }
        if(exists){
            return false;
        }else{
            addClassPathEntry( rootElement, exclusion, path);
            return true;
        }  
    }
    
    /**
     * Adds a unique class path entry.
     *
     * @param root the root
     * @param exclusion the exclusion
     * @param path the path
     */
    protected void addClassPathEntry(Element root,String exclusion,String path){
        Element newClasspathEntry = new Element("classpathentry");
        root.addContent(newClasspathEntry);
        newClasspathEntry.setAttribute("excluding", exclusion);
        newClasspathEntry.setAttribute("kind", "src");
        newClasspathEntry.setAttribute("path", path);
        logger.debug("" + newClasspathEntry);
    }
    
    /**
     * Save xml.
     * 
     * @param doc
     *            the doc
     * @param xmlFile
     *            the xml file
     */
    private void saveXML(Document doc, File xmlFile) {
        XMLOutputter xmlOutput = new XMLOutputter();
        xmlOutput.setFormat(Format.getPrettyFormat());
        SxmFileUtils.writeXMLToFile(xmlOutput, doc, xmlFile.getAbsolutePath());
    }
    
    /**
     * Gets the project name.
     *
     * @return the project name
     */
    public String getProjectName() {
        return projectName;
    }
    
    /**
     * Sets the project name.
     *
     * @param projectName the new project name
     */
    public void setProjectName(String projectName) {
        this.projectName = projectName;
    }
    
}
