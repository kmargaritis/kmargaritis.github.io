package org.jsxm.maven.plugin.jsxmtool.core;

import java.util.concurrent.ExecutorService;
import java.util.concurrent.Executors;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * @author Konstantinos Margaritis
 *
 */
public final class JSXMThreadPoolFactory {
    
    private final Logger logger = (Logger) LoggerFactory.getLogger(this.getClass());
     
    private int userThreads;
    private String banimationThreadPool; 
    private String compilationThreadPool;
    private String sampleCreationThreadPool;
    private String externalPackageCreationThreadPool;
    private String generationThreadPool;
    private String transformationThreadPool;
    private String validationThreadPool;
    
    
    
    private JSXMThreadPoolFactory() {
    }
    
    /**
     * The Class LazyHolder.
     */
    private static class LazyHolder {
        private static final JSXMThreadPoolFactory INSTANCE = new JSXMThreadPoolFactory();
    }
    
    /**
     * Gets the single instance of JSXMFacade.
     * 
     * @return single instance of JSXMFacade
     */
    public static JSXMThreadPoolFactory getInstance() {
        return LazyHolder.INSTANCE;
    }
    
    
    /**
     * Gets the executor.
     *
     * @return the executor
     */
    public synchronized ExecutorService getExecutor(String executor){
        ExecutorService jsxmExecutor;
        if (executor.equals("fixedThreadPool") || executor.equals("cachedThreadPool") ) {
            jsxmExecutor = getMultiThreadImplementation(executor);
        } else{
            jsxmExecutor=getSingleThreadImplementation(executor);
        } 
        return jsxmExecutor;
    }
    
    /**
     * Gets the multi thread implementation.
     *
     * @return the multi thread implementation
     */
    private synchronized ExecutorService getMultiThreadImplementation(String executor){
        ExecutorService jsxmCachedExecutor;
        if(executor.equals("fixedThreadPool")){
            if(userThreads>0){
                jsxmCachedExecutor=Executors.newFixedThreadPool(userThreads);   
                logger.info("---  Multithreading with newFixedThreadPool of "+ userThreads+ " threads.");
            }else{
                jsxmCachedExecutor=Executors.newFixedThreadPool(JSXM.THREADS);   
                logger.info("---  Multithreading with newFixedThreadPool of"+ JSXM.THREADS+ " threads.");
            }
        }
        else{
            jsxmCachedExecutor=Executors.newCachedThreadPool();
            logger.info("---  Multithreading with newCachedThreadPool.");
        }
        
        return jsxmCachedExecutor;
    }

    /**
     * Gets the single thread implementation.
     *
     * @return the single thread implementation
     */
    private synchronized ExecutorService getSingleThreadImplementation(String executor){
        ExecutorService selectedExecutor;
        if(executor.equals("singleThreadExecutor")){
            selectedExecutor = Executors.newSingleThreadExecutor();
            logger.info("---  Single Threading with newSingleThreadExecutor.");
        }
        else{
            selectedExecutor = Executors.newFixedThreadPool(1);
            logger.info("---  Single Threading with newFixedThreadPool with one thread.");
        }
        return selectedExecutor;
    }
    
    /**
     * @return the banimationThreadPool
     */
    public String getBanimationThreadPool() {
        return banimationThreadPool;
    }

    /**
     * @param banimationThreadPool the banimationThreadPool to set
     */
    public void setBanimationThreadPool(String banimationThreadPool) {
        this.banimationThreadPool = banimationThreadPool;
    }

    /**
     * @return the compilationThreadPool
     */
    public String getCompilationThreadPool() {
        return compilationThreadPool;
    }

    /**
     * @param compilationThreadPool the compilationThreadPool to set
     */
    public void setCompilationThreadPool(String compilationThreadPool) {
        this.compilationThreadPool = compilationThreadPool;
    }

    /**
     * @return the sampleCreationThreadPool
     */
    public String getSampleCreationThreadPool() {
        return sampleCreationThreadPool;
    }

    /**
     * @param sampleCreationThreadPool the sampleCreationThreadPool to set
     */
    public void setSampleCreationThreadPool(String sampleCreationThreadPool) {
        this.sampleCreationThreadPool = sampleCreationThreadPool;
    }

    /**
     * @return the externalPackageCreationThreadPool
     */
    public String getExternalPackageCreationThreadPool() {
        return externalPackageCreationThreadPool;
    }

    /**
     * @param externalPackageCreationThreadPool the externalPackageCreationThreadPool to set
     */
    public void setExternalPackageCreationThreadPool(String externalPackageCreationThreadPool) {
        this.externalPackageCreationThreadPool = externalPackageCreationThreadPool;
    }

    /**
     * @return the generationThreadPool
     */
    public String getGenerationThreadPool() {
        return generationThreadPool;
    }

    /**
     * @param generationThreadPool the generationThreadPool to set
     */
    public void setGenerationThreadPool(String generationThreadPool) {
        this.generationThreadPool = generationThreadPool;
    }

    /**
     * @return the transformationThreadPool
     */
    public String getTransformationThreadPool() {
        return transformationThreadPool;
    }

    /**
     * @param transformationThreadPool the transformationThreadPool to set
     */
    public void setTransformationThreadPool(String transformationThreadPool) {
        this.transformationThreadPool = transformationThreadPool;
    }

    /**
     * @return the validationThreadPool
     */
    public String getValidationThreadPool() {
        return validationThreadPool;
    }

    /**
     * @param validationThreadPool the validationThreadPool to set
     */
    public void setValidationThreadPool(String validationThreadPool) {
        this.validationThreadPool = validationThreadPool;
    }


    /**
     * @return the userThreads
     */
    public synchronized int getUserThreads() {
        return userThreads;
    }

    /**
     * @param userThreads the userThreads to set
     */
    public synchronized void setUserThreads(int userThreads) {
        this.userThreads = userThreads;
    }
    
}
