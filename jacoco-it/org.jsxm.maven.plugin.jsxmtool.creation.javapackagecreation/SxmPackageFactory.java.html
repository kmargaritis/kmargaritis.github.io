package org.jsxm.maven.plugin.jsxmtool.creation.javapackagecreation;

import java.io.File;
import java.util.List;

import org.jsxm.maven.plugin.jsxmtool.runtimeexceptions.SxmPackageCreatorUnknownImplementationException;

/**
 * @author Konstantinos Margaritis
 * 
 */
public class SxmPackageFactory {
    
    /**
     * Gets the creator.
     * 
     * @param creatorType
     *            the creator type
     * @param javaDirectory
     *            the java directory
     * @param externalPackageList
     *            the external package list
     * @return the creator
     */
    public SxmPackageCreator getCreator(PackageType packageType, File javaDirectory, List<String> externalPackageList) {
        SxmPackageCreator sxmPackageCreator = null;
        if (packageType==PackageType.INTERNAL) {
            if (!javaDirectory.getAbsolutePath().equals("")) {
                sxmPackageCreator = new SxmInternalPackageCreator();
                ((SxmInternalPackageCreator) sxmPackageCreator).setJavaDirectory(javaDirectory);
            }
        } else if (packageType==PackageType.EXTERNAL) {
            if (!externalPackageList.isEmpty()) {
                sxmPackageCreator = new SxmExternalPackageCreator();
                ((SxmExternalPackageCreator) sxmPackageCreator).setExternalPackageList(externalPackageList);
            }
        } else {
            throw new SxmPackageCreatorUnknownImplementationException("\nUnknown implementation for input " + packageType);
        }
        return sxmPackageCreator;
    }
}
