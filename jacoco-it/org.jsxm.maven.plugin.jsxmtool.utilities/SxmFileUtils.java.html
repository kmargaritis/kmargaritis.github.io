package org.jsxm.maven.plugin.jsxmtool.utilities;

import java.io.File;
import java.io.FileNotFoundException;
import java.io.FileOutputStream;
import java.io.IOException;
import java.io.OutputStreamWriter;
import java.nio.charset.Charset;
import java.util.List;

import org.apache.commons.io.FileUtils;
import org.apache.commons.io.filefilter.FileFileFilter;
import org.apache.commons.io.filefilter.FileFilterUtils;
import org.apache.commons.io.filefilter.IOFileFilter;
import org.jdom.Document;
import org.jdom.output.XMLOutputter;

/**
 * @author Konstantinos Margaritis
 * 
 */
public final class SxmFileUtils {
    
    private SxmFileUtils() {
    }
    
    /**
     * Given a list of strings creates the directories for all the lists
     * elements.
     * 
     * @param directoryNames
     *            the directory names
     */
    public static void makeDirectories(List<String> directoryNames) {
        try {
            for (int i = 0; i < directoryNames.size(); i++) {
                final String directory = directoryNames.get(i);
                File dir = new File(directory);
                if (dir.getParentFile().exists()) {
                    final boolean success = (new File(directory)).mkdir();
                    if (success) {
                        JSXMUtils.LOGGER.debug("Directory: " + directory + " created");
                    }
                } else {
                    final boolean success = (new File(directory)).mkdirs();
                    if (success) {
                        JSXMUtils.LOGGER.debug("Multiple Directories: " + directory + " created");
                    }
                }
            }
        } catch (Exception e) {
            JSXMUtils.LOGGER.error("Exception: Error in creating the directories : ", e.getMessage());
        }
    }
    
    /**
     * Sets the classpath for all the folders that needed to be known.
     * 
     * @param paths
     *            the new class path
     */
    public static void setClassPath(List<String> paths) {
        try {
            for (int i = 0; i < paths.size(); i++) {
                String path = JSXMUtils.PATH_SEPARATOR + paths.get(i);
                System.setProperty("java.class.path", System.getProperty("java.class.path") + path);
            }
            JSXMUtils.LOGGER.debug(System.getProperty("java.class.path"));
        } catch (Exception e) {
            JSXMUtils.LOGGER.error("Exception : ", e);
        }
    }
    
    /**
     * Traverse directory.
     * 
     * @param node
     *            the node
     * @param listOfFiles
     *            the list of files
     * @param extension
     *            the extension
     */
    public static void traverseDirectory(File node, List<File> listOfFiles, String extension) {
        if (node.isFile() && node.getName().endsWith(extension)) {
            listOfFiles.add(node);
        }
        if (node.isDirectory()) {
            String[] subNote = node.list();
            for (String filename : subNote) {
                traverseDirectory(new File(node, filename), listOfFiles, extension);
            }
        }
    }
    
    /**
     * Traverse directory with exclusion.
     * 
     * @param node
     *            the node
     * @param listOfFiles
     *            the list of files
     * @param extension
     *            the extension
     */
    public static void traverseDirectoryWithExclusion(File node, List<File> listOfFiles, String extension) {
        if (node.isFile() && node.getName().endsWith(extension) && !node.getName().endsWith("_sets.xml")
                && !node.getName().endsWith("_test.xml") && !node.getName().endsWith("_output.xml")
                && !node.getName().endsWith("_input.xml") && !node.getName().endsWith("definitions.xml")) {
            listOfFiles.add(node);
            JSXMUtils.LOGGER.debug("Node " + node + " added");
        }
        if (node.isDirectory()) {
            String[] subNote = node.list();
            for (String filename : subNote) {
                traverseDirectoryWithExclusion(new File(node, filename), listOfFiles, extension);
            }
        }
    }
    
    /**
     * Ignore file.
     *
     * @param file the file
     * @param ignoreList the ignore list
     * @return true, if successful
     */
    public static boolean ignoreFile(File file, String[] ignoreList) {
        for (final String ignoreStr : ignoreList){
            if (file.getAbsolutePath().contains(ignoreStr)){
                return true;
            }
        }
           
        return false;
    }
    
    /**
     * Copy a whole directory with all file and folders.
     * 
     * @param source
     *            the source
     * @param destination
     *            the destination
     */
    public static void copyDirectory(File source, File destination,String suffixFilter) {
        try {
            if (source.exists()) {
                IOFileFilter javaSuffixFilter = FileFilterUtils.suffixFileFilter(suffixFilter);
                IOFileFilter txtFiles = FileFilterUtils.and(FileFileFilter.FILE, javaSuffixFilter);
                FileUtils.copyDirectory(source, destination,txtFiles);
                JSXMUtils.LOGGER.debug("Copying: " + source);
                JSXMUtils.LOGGER.debug("Destination: " + destination);
                JSXMUtils.LOGGER.debug("File from " + source.getName() + " was copied to " + destination.getName());
            }
        } catch (IOException e) {
            JSXMUtils.LOGGER.error("IOException : ", e);
        }
    }
    
    /**
     * Copy file to directory.
     * 
     * @param source
     *            the source
     * @param destination
     *            the destination
     */
    public static void copyFileToDirectory(File source, File destination) {
        try {
            if (source.exists()) {
                FileUtils.copyFileToDirectory(source, destination);
                JSXMUtils.LOGGER.debug("Copying: " + source);
                JSXMUtils.LOGGER.debug("Destination: " + destination);
                /*
                 * Tried NIO copy but produced many errors. 
                 * See revision 548.
                 */
                JSXMUtils.LOGGER.debug("File from " + source.getName() + " was copied to " + destination.getName());
            }
        } catch (IOException e) {
            JSXMUtils.LOGGER.error("File " + source + " could not be copied to " + destination);
            JSXMUtils.LOGGER.error("IOException : ", e);
        }
    }
    
    /**
     * Find path to parent.
     * 
     * @param file
     *            the file
     * @param specFolder
     *            the spec folder
     * @param path
     *            the path
     * @return the string builder
     */
    public static StringBuilder findPathToParent(File file, String specFolder, StringBuilder path) {
        String specificationExclusion = "specification";
        String animationExclusion = "animation";
        String testsExclusion = "tests";
        
        JSXMUtils.LOGGER.debug("Searching ---> root parent for " + file);
        if (file.getParentFile().getName().equals(specFolder)) {
            JSXMUtils.LOGGER.debug("Found file with spec parent : " + file);
            return path;
        } else if (file.getParentFile().getName().equals(specificationExclusion)
                || file.getParentFile().getName().equals(animationExclusion)
                || file.getParentFile().getName().equals(testsExclusion)) {
            return findPathToParent(new File(file.getParent()), specFolder, path);
        } else {
            JSXMUtils.LOGGER.debug("Recursive search --> ");
            return findPathToParent(new File(file.getParent()), specFolder,
                    path.insert(0, file.getParentFile().getName() + JSXMUtils.FILE_SEPARATOR));
        }
        
    }
    
    /**
     * Write xml to file.
     *
     * @param xmlOutput the xml output
     * @param doc the doc
     * @param fileToWrite the file to write
     */
    public static void writeXMLToFile(XMLOutputter xmlOutput, Document doc, String fileToWrite) {
        OutputStreamWriter outpuStream;
        try {
            outpuStream = new OutputStreamWriter(new FileOutputStream(fileToWrite), Charset.forName("UTF-8")
                    .newEncoder());
            xmlOutput.output(doc, outpuStream);
        } catch (FileNotFoundException e) {
            JSXMUtils.LOGGER.error("FileNotFoundException : ", e);
        } catch (IOException e) {
            JSXMUtils.LOGGER.error("IOException : ", e);
        }
    }
    
}
