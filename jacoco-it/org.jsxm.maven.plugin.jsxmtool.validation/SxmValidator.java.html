package org.jsxm.maven.plugin.jsxmtool.validation;

import java.io.File;
import java.io.IOException;

import javax.xml.XMLConstants;
import javax.xml.transform.Source;
import javax.xml.transform.stream.StreamSource;
import javax.xml.validation.Schema;
import javax.xml.validation.SchemaFactory;
import javax.xml.validation.Validator;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.slf4j.MDC;
import org.xml.sax.SAXException;

/**
 * @author Konstantinos Margaritis
 * 
 */
public class SxmValidator {
    
    private final Logger logger = (Logger) LoggerFactory.getLogger(this.getClass());
    private static final Object ADD_LOCK = new Object();
    
    /**
     * 
     */
    protected SxmValidator() {
    }
    
    /**
     * Validates the given xml file.
     * 
     * @param schemaFile
     *            the schema file
     * @param localSchemaFile
     *            the local schema file
     * @param xmlFile
     *            the xml file
     * @param schemaFactory
     *            the schema factory
     * @param schema
     *            the schema
     * @param validator
     *            the validator
     * @param temp
     *            the temp
     */
    public void validate(String localSchemaFile, File temp) {
        Source xmlFile = null;
        SchemaFactory schemaFactory = null;
        Schema schema = null;
        Validator validator = null;
        try {
            xmlFile = new StreamSource(new File(temp.getAbsoluteFile().toString()));
            logger.debug("Xml file for validation : " + xmlFile.toString() + " and file path used :"
                    + temp.getAbsoluteFile().toString());
            schemaFactory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
            schema = schemaFactory.newSchema(new StreamSource(SxmValidationManager.class
                    .getResourceAsStream(localSchemaFile)));
            validator = schema.newValidator();
            /*
             * Provided an XMLErrorHandler() to the validator to customize the
             * error messages. See revision 548.
             */
            validator.validate(xmlFile);
            synchronized (ADD_LOCK) {
                SxmValidationManager.getInstance().getSuccessFiles().add(temp);
            }
            logger.debug(xmlFile.getSystemId() + " is valid" + "\n");
            MDC.remove("specificationName");
        } catch (SAXException e) {
            synchronized (ADD_LOCK) {
                SxmValidationManager.getInstance().getFailedFiles().add(temp);
            }
            logger.error(xmlFile.getSystemId() + " is NOT valid");
            logger.error("Reason: " + e.getLocalizedMessage() + "\n");
            /*
             * SAXException was used in order to produce a different error See
             * revision 548.
             */
        } catch (IOException e) {
            synchronized (ADD_LOCK) {
                SxmValidationManager.getInstance().getFailedFiles().add(temp);
            }
            logger.error("IOException : ", e.getMessage());
        }
    }
    
}
