package org.jsxm.maven.plugin.mojo;

import java.io.File;
import java.util.Properties;

import org.apache.maven.plugin.AbstractMojo;
import org.apache.maven.plugin.MojoExecutionException;
import org.apache.maven.plugin.MojoFailureException;
import org.jsxm.maven.plugin.jsxmtool.JSXMFacade;

/**
 * @author Konstantinos Margaritis
 * 
 */

/**
 * Goal that compiles the JSXM specifications and transforms them
 * to Repast agents. The created java classes can be moved directly 
 * to the Repast project by providing the repastCopyDirectory property.  
 * 
 * @since Version 1.0
 * @goal compile-repast
 * @execute goal="validate"
 * @phase compile
 */
public class RepastCompilationMojo extends AbstractMojo {
    
    /**
     * Sets the maven local repository. By default the value is the default
     * value set by maven.
     * 
     * @parameter default-value="${settings.localRepository}"
     * @required
     */
    private String m2HomeRepast;
    
    /**
     * Sets the temp directory that temporary files from jsxm are going to be
     * saved. The default value is inside the target directory of a maven
     * project.
     * 
     * @parameter default-value="${project.build.directory}"
     * @required
     */
    private String tempTargetRepast;
    
    /**
     * Sets the test directory of the maven project. The default value is the
     * src/test/java of the maven project.
     * 
     * @parameter default-value="${basedir}/src/test/java/"
     * @required
     */
    private String testDirectoryRepast;
    
    /**
     * Sets the implementation directory of the maven project. The default value
     * is the src/main/java of the maven project
     * 
     * @parameter default-value="${basedir}/src/main/java/"
     * @required
     */
    private File javaDirectoryRepast;
    
    /**
     * Sets the directory for all the repast specifications. The default value
     * is the spec directory under src-->src/spec.
     * 
     * @parameter default-value="${basedir}/src/spec/"
     * @required
     */
    private String jsxmSpecRootDirRepast;
    
    /**
     * Sets the mode of the maven project. Verbose and overwrite are the
     * available options. By default the mode is set to overwrite
     * 
     * @parameter default-value="overwrite"
     * @required
     */
    private String modeRepast;
    
    /**
     * Sets the k value that is going to be used in the generation of tests. The
     * default value is 2.
     * 
     * @parameter default-value="2"
     * @required
     */
    private String kInputRepast;
    
    /**
     * Sets the directory which the [specName]SXM.java and
     * [specName]SXM_base.java are going to be copied. There is no default value
     * for this property. Omitting this property will cause build failure in the
     * compile-repast goal.
     * 
     * @parameter
     * @required
     */
    private String repastCopyDirectory;
    
    /**
     * Repast imports for the SXM.java. The default imports are import
     * org.jsxm.jsxmcore.core.*;\n import org.jsxm.jsxmcore.types.*;\n import
     * java.util.*;\n import repast.simphony.engine.schedule.ScheduledMethod;\n
     * import repast.simphony.random.*;\n import
     * repast.simphony.query.space.grid.*;\n import
     * repast.simphony.space.grid.*;\n import repast.simphony.space.*;\n import
     * repast.simphony.context.*;\n import repast.simphony.util.*;\n"
     * 
     * @parameter default-value=
     *            "import org.jsxm.jsxmcore.core.*;\nimport org.jsxm.jsxmcore.types.*;\nimport org.jsxm.jsxmcore.exceptions.*;\nimport java.util.*;\nimport repast.simphony.engine.schedule.ScheduledMethod;\nimport repast.simphony.random.*;\nimport repast.simphony.query.space.grid.*;\nimport repast.simphony.space.grid.*;\nimport repast.simphony.space.*;\nimport repast.simphony.context.*;\nimport repast.simphony.util.*;\n"
     * @required
     */
    private String repastSxmImports;
    
    /**
     * JSXM imports for the SXM_base.java. The default imports are import
     * org.jsxm.jsxmcore.core.*;\n import org.jsxm.spatial.SXMSpatial;\n
     * 
     * @parameter default-value=
     *            "import org.jsxm.jsxmcore.core.*;\nimport org.jsxm.jsxmcore.exceptions.*;\nimport org.jsxm.spatial.SXMSpatial;\n"
     * @required
     */
    private String repastSxmBaseImports;
    
    /**
     * The type that the SXM_base.java is inheriting. The default value is
     * SXMSpatial
     * 
     * @parameter default-value="SXMSpatial"
     * @required
     */
    private String repastSxmBaseInheritance;
    
    /**
     * List in order to provide different k for each specification. For example
     * if a specification A needs k=2 and specification B k=4 kList must
     * contain, &lt;param&gt;&lt;/param&gt;
     * 
     * 
     * The kList is a list of properties that allows the execution of examples
     * with a different k than the default. In case an example contains more
     * than one specification, then there is the possibility to need a different
     * k value for some of the specifications but at the same time not changing
     * the general k value. In order to use this tag the following configuration
     * must be followed. 
     *  &lt;kList>
     *      &lt;property>
     *          &lt;name&gt;&lt;/name&gt;
     *          &lt;value&gt;&lt;/value&gt;
     *      &lt;/property&gt;
     *  &lt;/kList&gt; 
     * Consider the example Book_Borrower which contains 3
     * specifications. \The specifications Book, Borrower and Library will use
     * the k value 2 in case none configuration is specified. On the other hand,
     * if you would like to execute the Library specification with a k=4, and
     * Book with k=5 then you should consider the following. 
     *  &lt;kList&gt;
     *      &lt;property&gt;
     *          &lt;name&gtLibrary&lt;/name&gt;
     *          &lt;value&gt4&lt;/value>&gt;
     *      &lt;/property&gt; 
     *      &lt;property&gt;
     *          &lt;name&gt;Book&lt;/name&gt; 
     *          &lt;value&gt5&lt;/value&gt; 
     *      &lt;/property&gt; 
     *  &lt;/kList&gt; The above setting
     * will override the default k and will use the specified instead. The
     * Borrower specification will use k=2. The name tag must be identical with
     * the specification name.
     * 
     * @parameter
     */
    private Properties kList;
    
    /**
     * The name of the project.
     * 
     * @parameter default-value="${project.name}"
     * @required
     */
    private String projectName;
    
    /**
     * Boolean for creating a pdf report of the log, system configuration and
     * user settings.
     * 
     * @parameter default-value="true"
     * @required
     */
    private boolean pdfReportFlag;
    
    /* (non-Javadoc)
     * @see org.apache.maven.plugin.Mojo#execute()
     */
    @Override
    public void execute() throws MojoExecutionException, MojoFailureException {        
        JSXMFacade.getInstance().compileRepastSpecfication(m2HomeRepast, tempTargetRepast, testDirectoryRepast,
                javaDirectoryRepast, jsxmSpecRootDirRepast, modeRepast, kInputRepast, repastSxmBaseImports,
                repastSxmImports, repastSxmBaseInheritance, kList, repastCopyDirectory);
        JSXMFacade.getInstance().createPDFReport(pdfReportFlag,projectName);
    }
}
