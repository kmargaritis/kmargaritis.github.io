package org.jsxm.maven.plugin.jsxmtool.animation.batchanimation;

import org.jsxm.maven.plugin.jsxmtool.core.Specification;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.slf4j.MDC;

/**
 * @author Konstantinos Margaritis
 * 
 */
public class SxmBanimationThread implements Runnable {
    
    private boolean silent;
    private Specification specification;
    private final Logger logger = (Logger) LoggerFactory.getLogger(this.getClass());
    private static final Object CONSTRUCTOR_LOCK = new Object();
    
    /**
     * Instantiates a new banimation thread.
     * 
     * @param silent
     *            the silent
     * @param specification
     *            the specification
     */
    public SxmBanimationThread(boolean silent, Specification specification) {
        this.silent = silent;
        this.specification = specification;
    }
    
    @Override
    public void run() {
        logger.debug("Threading for " + specification.getClassName());
        logger.debug(specification.getClassName());
        MDC.put("specificationName", specification.getClassName());
        SxmBanimator sxmBanimator;
        synchronized (CONSTRUCTOR_LOCK) {
            sxmBanimator = new SxmBanimator();
            logger.debug(" Thread created sxmbanimator " + sxmBanimator);
        }
        sxmBanimator.banimateSXM(specification.getClassName(), silent, specification.getMode(),
                specification.getAnimationDirectory(), specification.getTempDirectory(),
                specification.getSpecificationDirectory());
        MDC.remove("specificationName");
    }
    
}
