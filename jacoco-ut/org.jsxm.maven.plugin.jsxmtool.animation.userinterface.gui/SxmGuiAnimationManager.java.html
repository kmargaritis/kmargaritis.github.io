package org.jsxm.maven.plugin.jsxmtool.animation.userinterface.gui;

import java.awt.EventQueue;
import java.util.List;

import javax.swing.JFrame;
import javax.swing.UIManager;

import org.jsxm.maven.plugin.jsxmtool.core.Specification;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * @author Konstantinos Margaritis
 *
 */
public class SxmGuiAnimationManager {
    
    private static final int HIGH_RES=290;
    private static final int LOW_RES=150;
    private static final int MINIMUM_HEIGHT=1050;
    private final Logger logger = (Logger) LoggerFactory.getLogger(this.getClass());
    
    /**
     * Initiates the graphical animation.
     *
     * @param specificationList the specification list
     * @param animationDelay the animation delay
     */
    public void animate(List<Specification> specificationList,int animationDelay){
        int pretextAreaHeight=HIGH_RES;
        if(java.awt.Toolkit.getDefaultToolkit().getScreenSize().height < MINIMUM_HEIGHT){
            pretextAreaHeight=LOW_RES;
        }
        final int textAreaHeight=pretextAreaHeight;
        final List<Specification> guiSpecificationList=specificationList;
        final int delay=animationDelay;
        try {
            createMacView();
            UIManager.setLookAndFeel(UIManager.getSystemLookAndFeelClassName());
        } catch (ClassNotFoundException ex) {
            logger.error("" + ex);
        } catch (InstantiationException ex) {
            logger.error("" + ex);
        } catch (IllegalAccessException ex) {
            logger.error("" + ex);
        } catch (javax.swing.UnsupportedLookAndFeelException ex) {
            logger.error("" + ex);
        }
        EventQueue.invokeLater(new Runnable() {
            @Override
            public void run() {
                JFrame sxmAnimator = new SxmGraphical(guiSpecificationList,delay,textAreaHeight);
                sxmAnimator.pack();
                sxmAnimator.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
                sxmAnimator.setResizable(false);
                sxmAnimator.setLocationRelativeTo(null);
                sxmAnimator.setVisible(true);
            }
        });

    }
    
    /**
     * Creates the mac view.
     */
    private void createMacView(){
         System.setProperty("com.apple.mrj.application.apple.menu.about.name","JSXM Animator");
         System.setProperty("apple.laf.useScreenMenuBar", "false"); 
    }
    
}
