package org.jsxm.maven.plugin.jsxmtool.animation.userinterface.gui;

import java.io.UnsupportedEncodingException;

import org.jdom.Attribute;
import org.jdom.Document;
import org.jdom.Element;
import org.jdom.Namespace;
import org.jdom.output.XMLOutputter;
import org.jsxm.jsxmcore.core.SXM;
import org.jsxm.jsxmcore.util.Various;
import org.jsxm.maven.plugin.jsxmtool.animation.userinterface.console.SxmConsoleAnimator;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.slf4j.MDC;

/**
 * @author Konstantinos Margaritis
 * 
 */
public class SxmGuiAnimator extends SxmConsoleAnimator {
    
    private final Logger logger = (Logger) LoggerFactory.getLogger(this.getClass());
    
    private String className;
    private String animationDirectory;
    private String specificationDirectory;
    private Document doc;
    private XMLOutputter xmlOutput;
    private Element rootElement;
    
    /**
     * Inits the gui animation.
     *
     * @param className the class name
     * @param animationDirectory the animation directory
     * @param specificationDirectory the specification directory
     * @throws UnsupportedEncodingException the unsupported encoding exception
     */
    public void initGUIAnimation(String className, String animationDirectory, String specificationDirectory)
            throws UnsupportedEncodingException {
        MDC.put("specificationName", className);
        instantiateSXM(className, animationDirectory, specificationDirectory);
        rootElement = new Element("sequence", Various.getInstance().getNameSpace());
        Namespace ns = Namespace.getNamespace("xsi", "http://www.w3.org/2001/XMLSchema-instance");
        rootElement.addNamespaceDeclaration(ns);
        rootElement.setAttribute(new Attribute("schemaLocation",
                "http://www.jsxm.org/schema http://www.jsxm.org/schema/input.xsd", ns));
        xmlOutput = new XMLOutputter();
        doc = new Document(rootElement);
        logger.info("SXM Animator in interactive mode.\n");
        logger.info("Enter .h. to see a list of commands.\n");
        reset();
        setBufferedReader(animationInputSource("", "console",animationDirectory));
        this.className = className;
        this.specificationDirectory = specificationDirectory;
        this.animationDirectory = animationDirectory;
        MDC.remove("specificationName");
    }
        
    /**
     * Reset gui animator.
     */
    public void resetGuiAnimator(){
        rootElement = new Element("sequence", Various.getInstance().getNameSpace());
        Namespace ns = Namespace.getNamespace("xsi", "http://www.w3.org/2001/XMLSchema-instance");
        rootElement.addNamespaceDeclaration(ns);
        rootElement.setAttribute(new Attribute("schemaLocation",
                "http://www.jsxm.org/schema http://www.jsxm.org/schema/input.xsd", ns));
        xmlOutput = new XMLOutputter();
        doc = new Document(rootElement); 
    }
    
    public String getClassName() {
        return className;
    }
    
    public String getAnimationDirectory() {
        return animationDirectory;
    }
    
    public String getSpecificationDirectory() {
        return specificationDirectory;
    }
    
    public Document getDoc() {
        return doc;
    }
    
    public XMLOutputter getXmlOutput() {
        return xmlOutput;
    }
    
    public Element getRootElement() {
        return rootElement;
    }
    
    @Override
    protected void instantiateSXM(String className, String animationDirectory, String specificationDirectory) {
        super.instantiateSXM(className, animationDirectory, specificationDirectory);
        logger.debug("gui animator instantiate");
    }
    
    @Override
    protected SXM getFsm() {
        logger.debug("gui animator getFsm");
        return super.getFsm();
    }
}
