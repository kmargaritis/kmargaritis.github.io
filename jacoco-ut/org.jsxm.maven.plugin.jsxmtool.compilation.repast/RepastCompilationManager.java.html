package org.jsxm.maven.plugin.jsxmtool.compilation.repast;

import java.io.File;
import java.util.LinkedList;
import java.util.List;
import java.util.Properties;

import org.jsxm.maven.plugin.jsxmtool.compilation.sxm.SxmCompilationManager;
import org.jsxm.maven.plugin.jsxmtool.utilities.SxmFileUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * @author Konstantinos Margaritis
 * 
 */
public class RepastCompilationManager {
    
    private final Logger logger = (Logger) LoggerFactory.getLogger(this.getClass());
    
    private List<File> sxmList = new LinkedList<File>();
    private List<File> sxmBaseList = new LinkedList<File>();
    
    /**
     * Compile repast.
     * 
     * @param m2HomeRepast
     *            the m2 home repast
     * @param tempTargetRepast
     *            the temp target repast
     * @param testDirectoryRepast
     *            the test directory repast
     * @param javaDirectoryRepast
     *            the java directory repast
     * @param jsxmSpecRootDirRepast
     *            the jsxm spec root dir repast
     * @param modeRepast
     *            the mode repast
     * @param kInputRepast
     *            the k input repast
     * @param repastSxmBaseImports
     *            the repast sxm base imports
     * @param repastSxmImports
     *            the repast sxm imports
     * @param repastSxmBaseInheritance
     *            the repast sxm base inheritance
     * @param kList
     *            the k list
     */
    public void compileRepast(String m2HomeRepast, String tempTargetRepast, String testDirectoryRepast,
            File javaDirectoryRepast, String jsxmSpecRootDirRepast, String modeRepast, String kInputRepast,
            String repastSxmBaseImports, String repastSxmImports, String repastSxmBaseInheritance, Properties kList) {
        
        SxmCompilationManager sxmC = new SxmCompilationManager();
        sxmC.initialize(m2HomeRepast, tempTargetRepast, testDirectoryRepast, javaDirectoryRepast,
                jsxmSpecRootDirRepast, modeRepast, kInputRepast, repastSxmBaseImports, repastSxmImports,
                repastSxmBaseInheritance, kList);
        
        SxmFileUtils.traverseDirectory(new File(tempTargetRepast), sxmList, "SXM.java");
        SxmFileUtils.traverseDirectory(new File(tempTargetRepast), sxmBaseList, "SXM_base.java");
        
    }
    
    /**
     * Copy repast.
     * 
     * @param repastCopyDirectory
     *            the repast copy directory
     */
    public void copyRepast(String repastCopyDirectory) {
        StringBuffer sxmFilesBuffer = new StringBuffer();
        String separator=", ";
        for (int i = 0; i < sxmList.size(); i++) {
            SxmFileUtils.copyFileToDirectory(sxmList.get(i).getAbsoluteFile(), new File(repastCopyDirectory));
            sxmFilesBuffer.append(sxmList.get(i).getName() + separator);
        }
        String sxmFiles = sxmFilesBuffer.toString();
        
        if (!new File(repastCopyDirectory).exists()) {
            List<String> directoryNames = new LinkedList<String>();
            directoryNames.add(repastCopyDirectory);
            SxmFileUtils.makeDirectories(directoryNames);
        }
        
        StringBuffer sxmBaseFilesBuffer = new StringBuffer();
        for (int i = 0; i < sxmBaseList.size(); i++) {
            SxmFileUtils.copyFileToDirectory(sxmBaseList.get(i).getAbsoluteFile(), new File(repastCopyDirectory));
            sxmBaseFilesBuffer.append(sxmBaseList.get(i).getName() + separator);
        }
        String sxmBaseFiles = sxmBaseFilesBuffer.toString();
        
        logger.info("SXM files :" + sxmFiles + " moved to " + repastCopyDirectory);
        logger.info("SXMBase files :" + sxmBaseFiles + " moved to " + repastCopyDirectory);
        
    }
    
}
