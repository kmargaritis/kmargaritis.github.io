package org.jsxm.maven.plugin.jsxmtool.compilation.sxm.controllers;

import java.io.File;
import java.util.LinkedList;
import java.util.List;

import javax.tools.Diagnostic;
import javax.tools.JavaFileObject;

import org.jsxm.maven.plugin.jsxmtool.core.JSXM;
import org.jsxm.maven.plugin.jsxmtool.runtimeexceptions.SXMCompilationException;
import org.jsxm.maven.plugin.jsxmtool.utilities.JSXMUtils;
import org.jsxm.maven.plugin.jsxmtool.utilities.SxmCompilationUtils;
import org.jsxm.maven.plugin.jsxmtool.utilities.SxmFileUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * @author Konstantinos Margaritis
 * 
 */
public class SxmCompilationController {
    
    private final Logger logger = (Logger) LoggerFactory.getLogger(this.getClass());
    
    /**
     * Scan compilation errors.
     */
    public void scanCompilationErrors() {
        if (!JSXM.getInstance().getCompilationDiagnostics().isEmpty()) {
            for (int j = 0; j < JSXM.getInstance().getCompilationDiagnostics().size(); j++) {
                Diagnostic<?> diagnostic = JSXM.getInstance().getCompilationDiagnostics().get(j);
                File sxmFile = new File(((JavaFileObject) diagnostic.getSource()).getName());
                if (diagnostic.getKind().toString().equals("ERROR")) {
                    logger.error("\n");
                    logger.error("Compiler    --> " + diagnostic.getKind());
                    logger.error("Line Number --> " + diagnostic.getLineNumber());
                    logger.error("Message     -->" + diagnostic.getMessage(null));
                    logger.error("Source      -->" + sxmFile.getName() + "\n");
                } else {
                    logger.warn("Compiler     -->" + diagnostic.getKind());
                    logger.warn("Message     -->" + diagnostic.getMessage(null) + "\n");
                }
            }
        } else {
            logger.debug("Compiled without errors and warnings");
        }
    }
    
    /**
     * Given a directory it compiles all the SXM.java and SXM_base.java files in
     * that directory.
     * 
     * @param tempTarget
     *            the temp target
     */
    public void compileSxmJavaFiles(String tempTarget) {
        List<File> sxmNames = new LinkedList<File>();
        List<File> sxmBaseNames = new LinkedList<File>();
        List<File> names = new LinkedList<File>();
        List<String> uncompiledFiles = new LinkedList<String>();
        String tempDirectory = tempTarget + JSXMUtils.FILE_SEPARATOR + "temp" + JSXMUtils.FILE_SEPARATOR;
        logger.debug(tempDirectory);
        String typesDirectory = tempDirectory + "org" + JSXMUtils.FILE_SEPARATOR + "jsxm"
                + JSXMUtils.FILE_SEPARATOR + "jsxmcore" + JSXMUtils.FILE_SEPARATOR + "types"
                + JSXMUtils.FILE_SEPARATOR;
        logger.debug(typesDirectory);
        
        SxmFileUtils.traverseDirectory(new File(tempDirectory), sxmNames, "SXM.java");
        SxmFileUtils.traverseDirectory(new File(tempDirectory), sxmBaseNames, "SXM_base.java");
        names.addAll(sxmNames);
        logger.debug(sxmNames.toString());
        names.addAll(sxmBaseNames);
        logger.debug(sxmBaseNames.toString());
        
        if (!names.isEmpty()) {
            SxmCompilationUtils.compileListOfFiles(names, tempDirectory);
        }
        int extensionLenght = JSXM.JAVA_EXTENSION_LENGTH;
        String classExtension = ".class";
        for (int i = 0; i < names.size(); i++) {
            if (!new File(typesDirectory
                    + SxmCompilationUtils.extractSourceClass(names.get(i).getName(), extensionLenght) + classExtension)
                    .exists()) {
                uncompiledFiles.add(typesDirectory
                        + SxmCompilationUtils.extractSourceClass(names.get(i).getName(), extensionLenght)
                        + classExtension);
                logger.error("SXM Class " + typesDirectory
                        + SxmCompilationUtils.extractSourceClass(names.get(i).getName(), extensionLenght)
                        + classExtension + " has not be compiled.");
            }
        }
        if (!uncompiledFiles.isEmpty()) {
            scanCompilationErrors();
            throw new SXMCompilationException("\nSXMCompilationException: Compilation errors.");
        }
    }
    
}
