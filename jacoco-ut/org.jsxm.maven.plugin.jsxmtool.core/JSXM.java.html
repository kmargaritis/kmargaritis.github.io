    /**
     * Adds a specification.
     * 
     * @param p
     *            the specification
     */
    public synchronized void addSpecification(Specification p) {
        boolean exists = false;
        for (int i = 0; i < specificationList.size(); i++) {
            if (specificationList.get(i).getName().equals(p.getName())) {
                exists = true;
            }
        }
        if (!exists) {
            specificationList.add(p);
        }
    }
    
    /**
     * Gets the specification list.
     *
     * @return the specification list
     */
    public synchronized List<Specification> getSpecificationList() {
        return specificationList;
    }
    
    /**
     * Sets the specification list.
     *
     * @param specificationList the new specification list
     */
    public synchronized void setSpecificationList(List<Specification> specificationList) {
        this.specificationList = specificationList;
    }
