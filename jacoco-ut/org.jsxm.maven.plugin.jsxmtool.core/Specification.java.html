    /**
     * SxmBasefiles in the getSpecification method return specification xml. In
     * windows systems return a string with a single backward slash causes
     * illegal escape character. If the environment is windows backward slash is
     * replaced with 2 backward slashes.
     */
    /**
     * 
     */
    public void setSpecificationPath() {
        if (!fileSeparator.equals("/")) {
            this.specificationPath = specificationFile.getAbsolutePath().replace("\\", "\\\\");
        } else {
            this.specificationPath = specificationFile.getAbsolutePath();
        }
    }
