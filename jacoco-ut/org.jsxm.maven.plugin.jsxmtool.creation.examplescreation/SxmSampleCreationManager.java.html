package org.jsxm.maven.plugin.jsxmtool.creation.examplescreation;

import java.io.File;
import java.util.LinkedList;
import java.util.List;
import java.util.concurrent.ExecutorService;

import org.jsxm.maven.plugin.jsxmtool.core.JSXMThreadPoolFactory;
import org.jsxm.maven.plugin.jsxmtool.utilities.SxmFileUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * @author Konstantinos Margaritis
 * 
 */
public class SxmSampleCreationManager {
    
    private final Logger logger = (Logger) LoggerFactory.getLogger(this.getClass());
    private final String fileSeparator = System.getProperty("file.separator");
    private List<String> javaFolders = new LinkedList<String>();
    private List<File> javaFileFolders = new LinkedList<File>();
    private List<String> rootFolders = new LinkedList<String>();
    private List<String> subFolders = new LinkedList<String>();
    private String javaDirectory;
    private String jsxmRootDir;
    private String testDirectory;
    private List<String> jsxmList;
    
    /**
     * Creates the roots, subfolders, and calls the cerator .
     */
    public void create() {
        createRootFolders();
        createSubFolders();  
        ExecutorService createJsxmExecutor=JSXMThreadPoolFactory.getInstance().getExecutor(JSXMThreadPoolFactory.getInstance().getSampleCreationThreadPool());
        for (int i = 0; i < jsxmList.size(); i++) {
            Runnable createJsxmWorker = new SxmSampleCreationThread(i, jsxmList, rootFolders, javaFileFolders);
            createJsxmExecutor.execute(createJsxmWorker);
        }
        /*
         * This will make the executor accept no new threads and finish all
         * existing threads in queue
         */
        createJsxmExecutor.shutdown();
        /*
         * Waiting for threads to finish execution
         */
        while (!createJsxmExecutor.isTerminated()) {
            logger.debug("Processing creation threads");
        }
        logger.debug("Validation threads have completed successfully");
        
    }
    
    /**
     * Creates the root folders.
     */
    public void createRootFolders() {
        for (int i = 0; i < jsxmList.size(); i++) {
            rootFolders.add(jsxmRootDir + jsxmList.get(i));
            if (System.getProperty("file.separator").equals("/")) {
                logger.debug("JavaDirectory :" + javaDirectory
                        + jsxmList.get(i).replace(jsxmList.get(i).substring(jsxmList.get(i).lastIndexOf('/') + 1), ""));
                javaFolders.add(javaDirectory
                        + jsxmList.get(i).replace(jsxmList.get(i).substring(jsxmList.get(i).lastIndexOf('/') + 1), ""));
                javaFileFolders
                .add(new File(javaDirectory
                        + jsxmList.get(i).replace(
                                jsxmList.get(i).substring(jsxmList.get(i).lastIndexOf('/') + 1), "")));
            } else {
                logger.debug("JavaDirectory :" + javaDirectory
                        + jsxmList.get(i).replace(jsxmList.get(i).substring(jsxmList.get(i).lastIndexOf('\\') + 1), ""));
                javaFolders
                .add(javaDirectory
                        + jsxmList.get(i).replace(
                                jsxmList.get(i).substring(jsxmList.get(i).lastIndexOf('\\') + 1), ""));
                javaFileFolders
                .add(new File(javaDirectory
                        + jsxmList.get(i).replace(
                                jsxmList.get(i).substring(jsxmList.get(i).lastIndexOf('\\') + 1), "")));
            }
            
        }
        SxmFileUtils.makeDirectories(rootFolders);
        SxmFileUtils.makeDirectories(javaFolders);
    }
    
    /**
     * Creates the subfolders.
     */
    public void createSubFolders() {
        for (int i = 0; i < rootFolders.size(); i++) {
            subFolders.add(rootFolders.get(i) + fileSeparator + "animation" + fileSeparator);
            subFolders.add(rootFolders.get(i) + fileSeparator + "specification" + fileSeparator);
            subFolders.add(rootFolders.get(i) + fileSeparator + "testAdapters" + fileSeparator);
        }
        SxmFileUtils.makeDirectories(subFolders);
    }
    
    public String getJsxmRootDir() {
        return jsxmRootDir;
    }
    
    public void setJsxmRootDir(String jsxmRootDir) {
        this.jsxmRootDir = jsxmRootDir;
    }
    
    public String getJavaDirectory() {
        return javaDirectory;
    }
    
    public void setJavaDirectory(String javaDirectory) {
        this.javaDirectory = javaDirectory;
    }
    
    public String getTestDirectory() {
        return testDirectory;
    }
    
    public void setTestDirectory(String testDirectory) {
        this.testDirectory = testDirectory;
    }
    
    public List<String> getJsxmList() {
        return jsxmList;
    }
    
    public void setJsxmList(List<String> jsxmList) {
        this.jsxmList = jsxmList;
    }
    
    public List<File> getJavaFileFolders() {
        return javaFileFolders;
    }
    
    public void setJavaFileFolders(List<File> javaFileFolders) {
        this.javaFileFolders = javaFileFolders;
    }
    
    public List<String> getRootFolders() {
        return rootFolders;
    }
    
    public void setRootFolders(List<String> rootFolders) {
        this.rootFolders = rootFolders;
    }
}
