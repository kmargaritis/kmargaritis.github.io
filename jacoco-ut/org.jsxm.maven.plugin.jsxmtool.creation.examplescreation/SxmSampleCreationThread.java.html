package org.jsxm.maven.plugin.jsxmtool.creation.examplescreation;

import java.io.File;
import java.util.LinkedList;
import java.util.List;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.slf4j.MDC;

/**
 * @author Konstantinos Margaritis
 * 
 */
public class SxmSampleCreationThread implements Runnable {
    
    private final Logger logger = (Logger) LoggerFactory.getLogger(this.getClass());
    private List<String> jsxmList;
    private int specificationIdentifier;
    private List<String> rootFolders = new LinkedList<String>();
    private List<File> javaFileFolders = new LinkedList<File>();
    private static final Object CONSTRUCTOR_LOCK = new Object();
    
    /**
     * Instantiates a new sxm creation thread.
     * 
     * @param specificationIdentifier
     *            the specification identifier
     * @param jsxmList
     *            the jsxm list
     * @param rootFolders
     *            the root folders
     * @param javaFileFolders
     *            the java file folders
     */
    public SxmSampleCreationThread(int specificationIdentifier, List<String> jsxmList, List<String> rootFolders,
            List<File> javaFileFolders) {
        this.jsxmList = jsxmList;
        this.specificationIdentifier = specificationIdentifier;
        this.rootFolders = rootFolders;
        this.javaFileFolders = javaFileFolders;
    }
    
    @Override
    public void run() {
        String jsxmName = "";
        if (System.getProperty("file.separator").equals("/")) {
            jsxmName = jsxmList.get(specificationIdentifier).substring(
                    jsxmList.get(specificationIdentifier).lastIndexOf('/') + 1);
        } else {
            jsxmName = jsxmList.get(specificationIdentifier).substring(
                    jsxmList.get(specificationIdentifier).lastIndexOf('\\') + 1);
        }
        MDC.put("specificationName", jsxmName);
        SxmSampleCreator sxmSampleCreator;
        synchronized (CONSTRUCTOR_LOCK) {
            sxmSampleCreator = new SxmSampleCreator(jsxmList, rootFolders, javaFileFolders);
            logger.debug("The sxm creator object " + sxmSampleCreator);
        }
        sxmSampleCreator.createSpecificationTemplate(specificationIdentifier);
        sxmSampleCreator.createSetsTemplate(specificationIdentifier);
        sxmSampleCreator.createImplementation(specificationIdentifier);
        sxmSampleCreator.createAdapter(specificationIdentifier);
        MDC.remove("specificationName");
    }
    
}
