package org.jsxm.maven.plugin.jsxmtool.creation.javapackagecreation;

import java.io.File;

import org.jsxm.maven.plugin.jsxmtool.core.JSXM;
import org.jsxm.maven.plugin.jsxmtool.utilities.SxmCompilationUtils;
import org.jsxm.maven.plugin.jsxmtool.utilities.SxmFileUtils;

/**
 * @author Konstantinos Margaritis
 * 
 */
public class SxmInternalPackageCreator extends SxmPackageStructureCreator implements SxmPackageCreator {
    
    protected SxmInternalPackageCreator() {
        
    }
    
    private File javaDirectory;
    
    /**
     * Copy packages.
     * 
     * @param javaDirectory
     *            the java directory
     * @param baseDir
     *            the base dir
     */
    @Override
    public void createSpecificationsFromPackages(String baseDir) {
        traversePackages(javaDirectory, baseDir);
    }
    
    /**
     * Traverse packages.
     * 
     * @param javaDirectory
     *            the java directory
     * @param baseDir
     *            the base dir
     */
    private void traversePackages(File javaDirectory, String baseDir) {
        getLogger().info("Java directory: {}",javaDirectory);
        getLogger().info("Base directory: {}",baseDir);
        SxmFileUtils.traverseDirectory(javaDirectory, getJavaFiles(), ".java");
        getLogger().info("Java files found: {}",getJavaFiles());
        for (File file : getJavaFiles()) {
            getLogger().info("File parent folder from traversed package: {}",file.getParent());
            getLogger().info("Java directory absolute path: {}",javaDirectory.getAbsolutePath());
            String parent = file.getParent().replace(javaDirectory.getAbsolutePath(), "");
            getLogger().info("Parent folder from traversed package: {}",parent);
            String specificationName = SxmCompilationUtils.extractSourceClass(file.getName(),
                    JSXM.JAVA_EXTENSION_LENGTH);
            getLogger().info("Specification name from traversed java file: {}",specificationName);
            createSpecificationFoldersTemplates(specificationName, parent, baseDir);
        }
        if (!isPackageFlag()) {
            getLogger().info("Nothing to create.");
        }
    }
    
    public File getJavaDirectory() {
        return javaDirectory;
    }
    
    public void setJavaDirectory(File javaDirectory) {
        this.javaDirectory = javaDirectory;
    }
    
}
