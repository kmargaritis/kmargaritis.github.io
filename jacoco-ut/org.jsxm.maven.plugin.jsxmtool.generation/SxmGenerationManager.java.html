package org.jsxm.maven.plugin.jsxmtool.generation;


import java.util.concurrent.ExecutorService;

import org.jsxm.maven.plugin.jsxmtool.core.JSXM;
import org.jsxm.maven.plugin.jsxmtool.core.JSXMThreadPoolFactory;
import org.jsxm.maven.plugin.jsxmtool.core.Specification;
import org.jsxm.maven.plugin.jsxmtool.utilities.JSXMUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * @author Konstantinos Margaritis
 * 
 */
public class SxmGenerationManager {
    
    private final Logger logger = (Logger) LoggerFactory.getLogger(this.getClass());
    
    /**
     * Generate.
     */
    public void generate() {
        long generatePhaseStart = System.currentTimeMillis();
        ExecutorService generateJsxmExecutor=JSXMThreadPoolFactory.getInstance().getExecutor(JSXMThreadPoolFactory.getInstance().getGenerationThreadPool());  
        for (int i = 0; i < JSXM.getInstance().getSpecificationList().size(); i++) {
            Specification p = JSXM.getInstance().getSpecificationList().get(i);
            Runnable generateJsxmWorker = new SxmGenerationThread(p);
            generateJsxmExecutor.execute(generateJsxmWorker);
        }
        /*
         * This will make the executor accept no new threads and finish all
         * existing threads in queue
         */
        generateJsxmExecutor.shutdown();
        /*
         * Waiting for threads to finish execution
         */
        long generationStart = System.currentTimeMillis();
        boolean flag = true;
        while (!generateJsxmExecutor.isTerminated()) {
            long current = System.currentTimeMillis();
            if (current - generationStart > JSXM.GENERATION_DELAY && flag) {
                logger.info("Processing..........\n");
                flag = false;
            }
        }     
        JSXMUtils.pdfReport(JSXM.getInstance().isPdfReport(), ".txt");
        long generatePhaseEnd = System.currentTimeMillis();
        long generatePhase=generatePhaseEnd-generatePhaseStart;
        logger.info("---  Generation phase done : "+generatePhase +" ms\n");
        /*
         * See revision 548 for resetting classpath.
         */
    }
}
