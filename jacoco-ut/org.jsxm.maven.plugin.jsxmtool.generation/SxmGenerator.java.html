package org.jsxm.maven.plugin.jsxmtool.generation;

import java.io.File;

import org.jsxm.jsxmcore.exceptions.DefinitionNotFoundException;
import org.jsxm.jsxmcore.testcasegenerators.SXMTestGeneration;
import org.jsxm.maven.plugin.jsxmtool.core.Specification;
import org.jsxm.maven.plugin.jsxmtool.runtimeexceptions.SXMCompilationException;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * @author Konstantinos Margaritis
 * 
 */
public class SxmGenerator {
    
    private final Logger logger = (Logger) LoggerFactory.getLogger(this.getClass());
    private static final Object GENERATE_LOCK = new Object();
    private final SXMTestGeneration testGen = new SXMTestGeneration();
    
    /**
     * Instantiates a new sxm generator.
     */
    public SxmGenerator() {
        
    }
    
    /**
     * Generate.
     * 
     * @param p
     *            the specification
     * @throws DefinitionNotFoundException
     *             the definition not found exception
     */
    public void generate(Specification p) throws DefinitionNotFoundException {
        String setsFileName = p.getSpecificationDirectory() + p.getClassName() + "_sets.xml";
        String xmlFileName = p.getTestsDirectory() + p.getClassName() + "_test.xml";
        String classFile = "org.jsxm.jsxmcore.types." + p.getClassName() + "SXM";
        generateTests(setsFileName, xmlFileName, classFile, p);
    }
    
    /**
     * Generate tests.
     * 
     * @param setsFileName
     *            the sets file name
     * @param xmlFileName
     *            the xml file name
     * @param classFile
     *            the class file
     * @param p
     *            the p
     * @throws DefinitionNotFoundException
     *             the definition not found exception
     */
    private void generateTests(String setsFileName, String xmlFileName, String classFile, Specification p)
            throws DefinitionNotFoundException {
        int k = Integer.parseInt(p.getkInput());
        if (p.getMode().equals("verbose")) {
            synchronized (GENERATE_LOCK) {
                logger.info("verbose mode on");
                logger.info("k = " + k);
                logger.info("sets  = " + setsFileName);
                logger.info("XML output file  = " + xmlFileName);
            }
        }
        if (new File(p.getTypesDirectory() + p.getClassName() + "SXM.class").exists()) {
            logger.debug("" + testGen);
            testGen.generateTests(k, setsFileName, xmlFileName, classFile);
            synchronized (GENERATE_LOCK) {
                logger.info("           -->SXM     Class : " + p.getClassName() + "SXM.java");
                logger.info("           -->SXMBase Class : " + p.getClassName() + "SXM_base.java");
                logger.info("           ###" + p.getClassName() +" tests generated successfully###\n");
            }
        } else {
            synchronized (GENERATE_LOCK) {
                logger.error("SXM Class " + p.getTypesDirectory() + p.getClassName() + "SXM.class"
                        + " has not been compiled.");
                logger.error("Test Generation for " + xmlFileName + " cancelled.");
                throw new SXMCompilationException("\nSXMCompilationException: Compilation errors.");
            }
        }
    }
    
}
