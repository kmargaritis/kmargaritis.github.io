package org.jsxm.maven.plugin.jsxmtool.logging.controllers;

import java.io.IOException;
import java.io.InputStream;

import org.jdom.JDOMException;
import org.jsxm.maven.plugin.jsxmtool.logging.logback.appenders.JsxmConsoleAppender;
import org.jsxm.maven.plugin.jsxmtool.logging.logback.filters.JsxmThresholdFilter;

/**
 * @author Konstantinos Margaritis
 * @since 1.3.4
 */
public class LogbackAppenderManager {
    
    private final Logger logger = LoggerFactory.getLogger(this.getClass());
    private String consoleLoggerLevel;
    private String htmlFileLoggerLevel;
    private String txtFileLoggerLevel;
    private boolean ansiSupport;
    
    /**
     * Sets the global loggers.
     */
    public void setGlobalLoggers() {
        createLogbackConfiguration();
        final LoggerContext lc = (LoggerContext) LoggerFactory.getILoggerFactory();
        Logger core = lc.getLogger("org.jsxm.maven.plugin.core");
        JsxmConsoleAppender<ILoggingEvent> console = (JsxmConsoleAppender<ILoggingEvent>) ((ch.qos.logback.classic.Logger) core)
                .getAppender("STDOUT");
        if (consoleLoggerLevel.equals("DEBUG")) {
            /*
             * See revision 604
             */
            StatusPrinter.print(lc);
        }
        if (ansiSupport) {
            console.setEncoder(getPatternLayoutEncoder(lc,  "%highlight(%-5level) - %blue(%msg%n)"));
            console.setWithJansi(true);
            console.start();
        }
    }
    
    /**
     * Creates the logback configuration.
     */
    private void createLogbackConfiguration() {
        LoggerContext lc = (LoggerContext) LoggerFactory.getILoggerFactory();
        try {
            JoranConfigurator configurator = new JoranConfigurator();
            configurator.setContext(lc);
            lc.reset();
            configurator.doConfigure(createLogbackXML());
        } catch (JoranException je) {
            logger.error("JoranException" + je);
        } catch (JDOMException e) {
            logger.error("JDOMException" + e);
        } catch (IOException e) {
            logger.error("IOException" + e);
        }
    }
    
    /**
     * Creates the logback xml.
     *
     * @return the input stream
     * @throws JDOMException the jDOM exception
     * @throws IOException Signals that an I/O exception has occurred.
     */
    private InputStream createLogbackXML() throws JDOMException, IOException {
        String configuration = LogbackConfigurationGenerator.generate(consoleLoggerLevel, htmlFileLoggerLevel,
                txtFileLoggerLevel).toString();
        return new ByteArrayInputStream(configuration.getBytes(Charset.forName("UTF-8")));        
    }
    
    /**
     * Replace jsxm threshold filter.
     * Use setGlobalLoggers() instead.
     */
    @Deprecated
    public void replaceJsxmThresholdFilter() {
        Level consoleLevel = Level.toLevel(consoleLoggerLevel);
        Level htmlLevel = Level.toLevel(htmlFileLoggerLevel);
        Level txtLevel = Level.toLevel(txtFileLoggerLevel);
        
        final LoggerContext lc = (LoggerContext) LoggerFactory.getILoggerFactory();
        
        if (consoleLoggerLevel.equals("DEBUG")) {
            /*
             * See revision 604
             */
            /*
             * Print logbacks internal status for debugging purposes.
             */
            StatusPrinter.print(lc);
        }
        
        Logger core = lc.getLogger("org.jsxm.maven.plugin.core");
        JsxmConsoleAppender<ILoggingEvent> console = (JsxmConsoleAppender<ILoggingEvent>) ((ch.qos.logback.classic.Logger) core)
                .getAppender("STDOUT");
        
        Logger root = lc.getLogger(Logger.ROOT_LOGGER_NAME);
        SiftingAppender sift = (SiftingAppender) ((ch.qos.logback.classic.Logger) root).getAppender("SIFT");
        SiftingAppender siftHtml = (SiftingAppender) ((ch.qos.logback.classic.Logger) root).getAppender("SIFT-HTML");
        
        for (int i = 0; i < console.getCopyOfAttachedFiltersList().size(); i++) {
            if (console.getCopyOfAttachedFiltersList().get(i).getClass().getName()
                    .equals("org.jsxm.maven.plugin.logback.filters.JsxmThresholdFilter")) {
                ((JsxmThresholdFilter) console.getCopyOfAttachedFiltersList().get(i)).setLevel(consoleLevel);
                logger.debug(console.getCopyOfAttachedFiltersList().get(i).getClass().toString());
            }
        }
        
        for (int i = 0; i < sift.getCopyOfAttachedFiltersList().size(); i++) {
            if (sift.getCopyOfAttachedFiltersList().get(i).getClass().getName()
                    .equals("org.jsxm.maven.plugin.logback.filters.JsxmThresholdFilter")) {
                ((JsxmThresholdFilter) sift.getCopyOfAttachedFiltersList().get(i)).setLevel(txtLevel);
                logger.debug(sift.getCopyOfAttachedFiltersList().get(i).getClass().toString());
            }
        }
        
        for (int i = 0; i < siftHtml.getCopyOfAttachedFiltersList().size(); i++) {
            if (siftHtml.getCopyOfAttachedFiltersList().get(i).getClass().getName()
                    .equals("org.jsxm.maven.plugin.logback.filters.JsxmThresholdFilter")) {
                ((JsxmThresholdFilter) siftHtml.getCopyOfAttachedFiltersList().get(i)).setLevel(htmlLevel);
                logger.debug(siftHtml.getCopyOfAttachedFiltersList().get(i).getClass().toString());
            }
        }
        
        if (ansiSupport) {
            console.setEncoder(getPatternLayoutEncoder(lc,  "%highlight(%-5level) - %blue(%msg%n)"));
            console.setWithJansi(true);
            console.start();
        }
        
        logger.debug("" + console.getCopyOfAttachedFiltersList());
        logger.debug("" + sift.getCopyOfAttachedFiltersList());
        logger.debug("" + siftHtml.getCopyOfAttachedFiltersList());
    }
    
    /**
     * Creates the console appender.
     *
     * @param consoleLevel the console level
     */
    @Deprecated
    public void createConsoleAppender(String consoleLevel) {
        LoggerContext context = (LoggerContext) LoggerFactory.getILoggerFactory();
        Logger core = (ch.qos.logback.classic.Logger) LoggerFactory.getLogger("org.jsxm.maven.plugin.jsxmtool");
        ((ch.qos.logback.classic.Logger) core).setAdditive(true);
        
        JsxmConsoleAppender<ILoggingEvent> jsFAppender = new JsxmConsoleAppender<>();
        jsFAppender.setContext(context);
        jsFAppender.setWithJansi(true);
        jsFAppender.setName("STDOUT");
        
        PatternLayoutEncoder encoder = new PatternLayoutEncoder();
        encoder.setContext(context);
        encoder.setPattern("[%-5level] [%d{HH:mm:ss.SSS}] %msg%n ");
        encoder.start();
        jsFAppender.setEncoder(encoder);
        
        if (consoleLevel.equals("DEBUG")) {
            JsxmThresholdFilter jFilter = new JsxmThresholdFilter();
            jFilter.setContext(context);
            jFilter.setLevel(Level.DEBUG);
            jsFAppender.addFilter(jFilter);
            jFilter.start();
        } else if (consoleLevel.equals("INFO")) {
            JsxmThresholdFilter jFilter = new JsxmThresholdFilter();
            jFilter.setContext(context);
            jFilter.setLevel(Level.INFO);
            jsFAppender.addFilter(jFilter);
            jFilter.start();
        } else if (consoleLevel.equals("WARN")) {
            JsxmThresholdFilter jFilter = new JsxmThresholdFilter();
            jFilter.setContext(context);
            jFilter.setLevel(Level.WARN);
            jsFAppender.addFilter(jFilter);
            jFilter.start();
        } else if (consoleLevel.equals("ERROR")) {
            JsxmThresholdFilter jFilter = new JsxmThresholdFilter();
            jFilter.setContext(context);
            jFilter.setLevel(Level.ERROR);
            jsFAppender.addFilter(jFilter);
            jFilter.start();
        }
        jsFAppender.start();
        ((ch.qos.logback.classic.Logger) core).addAppender(jsFAppender);
    }
            
    
    /**
     * Gets the pattern layout encoder.
     *
     * @param context the context
     * @param pattern the pattern
     * @return the pattern layout encoder
     */
    public PatternLayoutEncoder getPatternLayoutEncoder(LoggerContext context,String pattern) {
        PatternLayoutEncoder encoder = new PatternLayoutEncoder();
        encoder.setContext(context);
        encoder.setPattern(pattern);
        encoder.start();
        return encoder;
    }

    }
