package org.jsxm.maven.plugin.jsxmtool.logging.controllers;

import java.io.IOException;

import org.jsxm.jsxmcore.util.Various;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * The Class LogbackConfigurationGenerator.
 *
 * @author Konstantinos Margaritis
 */
public final class LogbackConfigurationGenerator {
    
    private LogbackConfigurationGenerator(){
        
    }
    
    /** The Constant logger. */
    private static  final Logger LOGGER = (Logger) LoggerFactory.getLogger(LogbackConfigurationGenerator.class);
    
    /**
     * Creates the base string.
     *
     * @return the string builder
     */
    private static StringBuilder createBaseString() {
        String file = "/templates/NEWLogbackConfiguration.template";
        StringBuilder fileContents = null;
        try {
            fileContents = Various.readFileFromJAR(file);
        } catch (IOException e) {
            LOGGER.error("Problem occured while trying to read file: " + file);
            LOGGER.error("IOException : ",e);
        }
        return fileContents;
    }

    /**
     * Generate.
     *
     * @param consoleLevel the console level
     * @param txtLevel the txt level
     * @param htmlLevel the html level
     * @return the string builder
     */
    public static StringBuilder generate(String consoleLevel,String htmlLevel,String txtLevel) {
        StringBuilder logbackConfiguration=createBaseString();
        org.jsxm.jsxmcore.util.Various.replaceString(logbackConfiguration, "NEW_CONSOLE_LEVEL", consoleLevel);
        org.jsxm.jsxmcore.util.Various.replaceString(logbackConfiguration, "NEW_TXT_LEVEL", txtLevel);
        org.jsxm.jsxmcore.util.Various.replaceString(logbackConfiguration, "NEW_HTML_LEVEL", htmlLevel);
        return logbackConfiguration;
    }
    
}
