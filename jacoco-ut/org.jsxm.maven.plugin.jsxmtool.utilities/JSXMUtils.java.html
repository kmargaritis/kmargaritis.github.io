package org.jsxm.maven.plugin.jsxmtool.utilities;

import java.io.BufferedReader;
import java.io.File;
import java.io.FileNotFoundException;
import java.io.FileOutputStream;
import java.io.IOException;
import java.lang.reflect.InvocationTargetException;
import java.nio.charset.StandardCharsets;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.util.Date;
import java.util.LinkedList;
import java.util.List;

import org.jsxm.maven.plugin.jsxmtool.core.JSXM;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.slf4j.MDC;

import com.itextpdf.text.Document;
import com.itextpdf.text.DocumentException;
import com.itextpdf.text.Font;
import com.itextpdf.text.Paragraph;
import com.itextpdf.text.pdf.PdfWriter;

/**
 * @author Konstantinos Margaritis
 * 
 */
public final class JSXMUtils {
    
    private JSXMUtils() {
    }
    
    public static final Logger LOGGER = (Logger) LoggerFactory.getLogger(JSXMUtils.class);
    public static final String FILE_SEPARATOR = System.getProperty("file.separator");
    public static final String PATH_SEPARATOR = System.getProperty("path.separator");
    
    private static final Font SMALL_BOLD = new Font(Font.FontFamily.TIMES_ROMAN, 12, Font.BOLD);
    private static final Font CAT_FOND = new Font(Font.FontFamily.TIMES_ROMAN, 18, Font.BOLD);
    private static final Font SMALL_FOND = new Font(Font.FontFamily.TIMES_ROMAN, 6, Font.NORMAL);
    
    /**
     * Reset class path.
     */
    public static void resetClassPath() {
        try {
            
            ClassPathUpdater.add(JSXM.getInstance().getInitialPath());
            System.setProperty("java.class.path", JSXM.getInstance().getInitialPath());
        } catch (IOException e) {
            LOGGER.error("IOException : ", e);
        } catch (NoSuchMethodException e) {
            LOGGER.error("NoSuchMethodException : ", e);
        } catch (IllegalAccessException e) {
            LOGGER.error("IllegalAccessException : ", e);
        } catch (InvocationTargetException e) {
            LOGGER.error("InvocationTargetException : ", e);
        }
    }
    
    /**
     * Replace slash with dot.
     * 
     * @param beforeReplacement
     *            the previous string
     * @return the new string with the replaced character
     */
    public static String replaceSlashWithDot(String beforeReplacement) {
        String afterReplacement = "";
        if (System.getProperty("file.separator").equals("/")) {
            afterReplacement = beforeReplacement.replace('/', '.');
        } else {
            afterReplacement = beforeReplacement.replace('\\', '.');
        }
        return afterReplacement;
    }
    
    /**
     * Replace dot with slash.
     * 
     * @param beforeReplacement
     *            the previous string
     * @return the new string with the replaced character
     */
    public static String replaceDotWithSlash(String beforeReplacement) {
        String afterReplacement = "";
        if (System.getProperty("file.separator").equals("/")) {
            afterReplacement = beforeReplacement.replace('.', '/');
        } else {
            afterReplacement = beforeReplacement.replace('.', '\\');
        }
        return afterReplacement;
    }
    
    /**
     * Read txt write to pdf.
     * 
     * @param readFile
     *            the read file
     * @param outputFile
     *            the output file
     */
    public static void readTxtWriteToPdf(String readFile, String outputFile, String specificationName) {
        Path path = Paths.get(readFile).toAbsolutePath();
        // TODO: Add listener for logback output
        LOGGER.info("Txt path : " + path);
        // TODO: Add listener for logback output
        Path outputPath = Paths.get(outputFile).toAbsolutePath();
        LOGGER.info("Pdf path : " + outputPath);
        StringBuilder strB = new StringBuilder();
        Document document = new Document();
        
        try {
            PdfWriter.getInstance(document, new FileOutputStream(outputPath.toFile()));
            document.open();
            addMetaData(document, specificationName);
            Paragraph preface = new Paragraph();
            addEmptyLine(preface, 1);
            preface.add(new Paragraph(JSXM.getInstance().getProjectName() + " logs ", CAT_FOND));
            addEmptyLine(preface, 1);
            preface.add(new Paragraph("Report generated by: " + System.getProperty("user.name") + ", " + new Date(),
                    SMALL_BOLD));
            addEmptyLine(preface, 1);
            preface.add(new Paragraph("This document describes all the info, warning and error logs of "
                    + specificationName, SMALL_BOLD));
            addEmptyLine(preface, 1);
            preface.add(new Paragraph("System variables:", SMALL_BOLD));
            preface.add(new Paragraph("Operating System Name:  " + System.getProperty("os.name"), SMALL_FOND));
            preface.add(new Paragraph("Operating System Architecture:  " + System.getProperty("os.arch"), SMALL_FOND));
            preface.add(new Paragraph("Java version:  " + System.getProperty("java.version"), SMALL_FOND));
            
            document.add(preface);
            
            BufferedReader reader = Files.newBufferedReader(path, StandardCharsets.UTF_8);
            String line = null;
            while ((line = reader.readLine()) != null) {
                strB.append(line);
                document.add(new Paragraph(line, SMALL_FOND));
            }
            document.close();
        } catch (FileNotFoundException e) {
            LOGGER.error("File " + outputPath.toFile() + " was not found", e);
        } catch (DocumentException e) {
            LOGGER.error("Error creating pdf document.!", e);
        } catch (IOException e) {
            LOGGER.error("Error reading log.txt document.!", e);
        }
    }
    
    /**
     * Adds the meta data.
     * 
     * @param document
     *            the document
     */
    private static void addMetaData(Document document, String specificationName) {
        document.addTitle("Log for maven build");
        document.addKeywords("maven-jsxm-plugin," + JSXM.getInstance().getProjectName() + "," + specificationName);
        document.addAuthor(System.getProperty("user.name"));
        document.addCreator(System.getProperty("user.name"));
    }
    
    /**
     * Adds the empty line.
     * 
     * @param paragraph
     *            the paragraph
     * @param number
     *            the number
     */
    private static void addEmptyLine(Paragraph paragraph, int number) {
        for (int i = 0; i < number; i++) {
            paragraph.add(new Paragraph(" "));
        }
    }
    
    /**
     * Pdf report.
     * 
     * @param pdfReport
     *            the pdf report
     * @param extension
     *            the extension
     */
    public static void pdfReport(boolean pdfReport, String extension) {
        if (pdfReport) {
            File node = new File("src/spec/jsxmLogs/");
            List<File> txtFiles = new LinkedList<File>();
            SxmFileUtils.traverseDirectory(node, txtFiles, extension);
            
            for (int i = 0; i < txtFiles.size(); i++) {
                String mdcTxtName = SxmCompilationUtils.extractSourceClass(txtFiles.get(i).getName(),
                        JSXM.SPECIFICATION_EXTENSION_LENGTH);
                String txtName = SxmCompilationUtils.extractSourceClass(txtFiles.get(i).getAbsolutePath(),
                        JSXM.SPECIFICATION_EXTENSION_LENGTH);
                String pdfName = txtName + ".pdf";
                MDC.put("specificationName", mdcTxtName);
                readTxtWriteToPdf(txtFiles.get(i).getAbsolutePath(), pdfName, txtName);
                MDC.remove("specificationName");
            }
        }
    }
    
}
