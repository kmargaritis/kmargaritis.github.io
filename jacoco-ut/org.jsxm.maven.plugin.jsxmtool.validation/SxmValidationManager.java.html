package org.jsxm.maven.plugin.jsxmtool.validation;

import java.io.File;
import java.io.IOException;
import java.net.HttpURLConnection;
import java.net.MalformedURLException;
import java.net.URL;
import java.util.Date;
import java.util.LinkedList;
import java.util.List;
import java.util.ListIterator;
import java.util.concurrent.ExecutorService;

import org.jsxm.maven.plugin.jsxmtool.core.JSXM;
import org.jsxm.maven.plugin.jsxmtool.core.JSXMThreadPoolFactory;
import org.jsxm.maven.plugin.jsxmtool.runtimeexceptions.XMLValidationException;
import org.jsxm.maven.plugin.jsxmtool.utilities.SxmCompilationUtils;
import org.jsxm.maven.plugin.jsxmtool.utilities.SxmFileUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * @author Konstantinos Margaritis
 * 
 */
public final class SxmValidationManager {
    /**
     * 
     */
    private final Logger logger = (Logger) LoggerFactory.getLogger(this.getClass());
    
    private SxmValidationManager() {
    }
    
    /**
     * The Class LazyHolder.
     */
    private static class LazyHolder {
        private static final SxmValidationManager INSTANCE = new SxmValidationManager();
    }
    
    /**
     * Gets the single instance of SxmValidationManager.
     *
     * @return single instance of SxmValidationManager
     */
    public static SxmValidationManager getInstance() {
        return LazyHolder.INSTANCE;
    }
    
    /** The success files. */
    private List<File> successFiles = new LinkedList<File>();
    
    /** The failed files. */
    private List<File> failedFiles = new LinkedList<File>();
    
    /**
     * Validate jsxm.
     * 
     * @param jsxmSpecRootDir
     *            the jsxm spec root dir
     * @param testDirectory
     *            the test directory
     */
    public void validateJSXM(String jsxmSpecRootDir, String testDirectory) {
        long validatePhaseStart = System.currentTimeMillis();
        File specDirectory = new File(jsxmSpecRootDir);
        List<File> definitionsSetsSpecifications = new LinkedList<File>();
        SxmFileUtils.traverseDirectory(specDirectory, definitionsSetsSpecifications, ".xml");
        
        @SuppressWarnings({ "rawtypes", "unchecked" })
        ListIterator<File> iteratorSpecDef = (ListIterator) definitionsSetsSpecifications.iterator();
        validateSchemas(iteratorSpecDef);
        checkValidatedFiles(testDirectory, failedFiles);
        long validatePhaseEnd = System.currentTimeMillis();
        long validatePhase = validatePhaseEnd - validatePhaseStart;
        logger.info("---  Validating phase done : " + validatePhase + " ms");
    }
    
    /**
     * Check validated files.
     * 
     * @param testDirectory
     *            the test directory
     * @param failedFiles
     *            the failed files
     */
    public void checkValidatedFiles(String testDirectory, List<File> failedFiles) {
        if (!failedFiles.isEmpty()) {
            JSXM.getInstance().setValidate(false);
            deleteTestsOnFailure(failedFiles, testDirectory);
            logger.error("Validation ended with errors. Resolve errors in order to continue with compilation.");
            throw new XMLValidationException("\nValidation errors.");
        } else {
            JSXM.getInstance().setValidate(true);
            logger.info("---  Validation completed successfully.!");
        }
    }
    
    /**
     * Validate schemas.
     * 
     * @param iterator
     *            the iterator
     */
    private void validateSchemas(ListIterator<File> iterator) {
        ExecutorService validateJsxmExecutor = JSXMThreadPoolFactory.getInstance().getExecutor(JSXMThreadPoolFactory.getInstance().getValidationThreadPool());
        while (iterator.hasNext()) {
            File temp = iterator.next();
            Runnable validateJsxmWorker = new SxmValidationThread(temp);
            validateJsxmExecutor.execute(validateJsxmWorker);
        }
        /*
         * This will make the executor accept no new threads and finish all
         * existing threads in queue
         */
        validateJsxmExecutor.shutdown();
        /*
         * Waiting for threads to finish execution
         */
        while (!validateJsxmExecutor.isTerminated()) {
            logger.trace("Processing validation threads");
        }
    }
    
    /**
     * Checks connections to www.jsxm.org. Initially a HttpURLConnection is
     * opened and attempts to connect to www.jsxm.org. The connection has also
     * read and connect time out for preventing delays. The timeout time is 0.5
     * seconds. It returns false when the connection is successful and true
     * otherwise.
     * 
     * @return
     */
    @Deprecated
    public boolean checkConnection() {
        final URL url;
        final int timeOutTime = 500;
        try {
            url = new URL("http://www.jsxm.org");
            HttpURLConnection con = (HttpURLConnection) url.openConnection();
            con.setConnectTimeout(timeOutTime);
            con.setReadTimeout(timeOutTime);
            con.connect();
            logger.debug("Connection with " + url + " was successfull. Responce code: " + con.getResponseCode());
            return false;
        } catch (MalformedURLException e) {
            logger.debug("MalformedURLException at " + new Date(System.currentTimeMillis()).toString());
            logger.info("XSD schemas from www.jsxm.org are resolved locally.");
            return true;
        } catch (IOException e) {
            logger.debug("IOException at " + new Date(System.currentTimeMillis()).toString());
            logger.info("XSD schemas from www.jsxm.org are resolved locally.");
            return true;
        }
    }
    
    /**
     * Delete tests on failure.
     * 
     * @param specificationFolderList
     *            the specification folder list
     * @param testDirectory
     *            the test directory
     */
    public void deleteTestsOnFailure(List<File> specificationFolderList, String testDirectory) {
        for (int i = 0; i < specificationFolderList.size(); i++) {
            StringBuilder temp = SxmFileUtils.findPathToParent(specificationFolderList.get(i), "spec",
                    new StringBuilder());
            String className = "";
            
            if (SxmCompilationUtils.extractSourceClass(specificationFolderList.get(i).getName(),
                    JSXM.SPECIFICATION_EXTENSION_LENGTH).contains("_input")) {
                className = SxmCompilationUtils.extractSourceClass(specificationFolderList.get(i).getName(),
                        JSXM.SPECIFICATION_EXTENSION_LENGTH).replace("_input", "");
            } else if (SxmCompilationUtils.extractSourceClass(specificationFolderList.get(i).getName(),
                    JSXM.SPECIFICATION_EXTENSION_LENGTH).contains("_output")) {
                className = SxmCompilationUtils.extractSourceClass(specificationFolderList.get(i).getName(),
                        JSXM.SPECIFICATION_EXTENSION_LENGTH).replace("_output", "");
            } else if (SxmCompilationUtils.extractSourceClass(specificationFolderList.get(i).getName(),
                    JSXM.SPECIFICATION_EXTENSION_LENGTH).contains("_test")) {
                className = SxmCompilationUtils.extractSourceClass(specificationFolderList.get(i).getName(),
                        JSXM.SPECIFICATION_EXTENSION_LENGTH).replace("_test", "");
            } else if (SxmCompilationUtils.extractSourceClass(specificationFolderList.get(i).getName(),
                    JSXM.SPECIFICATION_EXTENSION_LENGTH).contains("_sets")) {
                className = SxmCompilationUtils.extractSourceClass(specificationFolderList.get(i).getName(),
                        JSXM.SPECIFICATION_EXTENSION_LENGTH).replace("_sets", "");
            } else {
                className = SxmCompilationUtils.extractSourceClass(specificationFolderList.get(i).getName(),
                        JSXM.SPECIFICATION_EXTENSION_LENGTH);
            }
            logger.debug("Checking file : " + testDirectory + temp + className + "JsxmAdapterTest.java");
            
            File tempFile = new File(testDirectory + temp + className + "JsxmAdapterTest.java");
            if (tempFile.exists()) {
                logger.error("Deleting test file" + tempFile + " due to validation errors");
                if (tempFile.delete()) {
                    logger.debug("File deleted");
                }
            }
        }
    }
    
    public List<File> getSuccessFiles() {
        return successFiles;
    }
    
    public void setSuccessFiles(List<File> successFiles) {
        this.successFiles = successFiles;
    }
    
    public List<File> getFailedFiles() {
        return failedFiles;
    }
    
    public void setFailedFiles(List<File> failedFiles) {
        this.failedFiles = failedFiles;
    }
}
