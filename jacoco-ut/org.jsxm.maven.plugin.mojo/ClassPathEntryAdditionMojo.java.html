package org.jsxm.maven.plugin.mojo;

import org.apache.maven.plugin.AbstractMojo;
import org.apache.maven.plugin.MojoExecutionException;
import org.apache.maven.plugin.MojoFailureException;
import org.jsxm.maven.plugin.jsxmtool.JSXMFacade;

/**
 * @author Konstantinos Margaritis
 *
 */

/**
 * Goal that adds all the specifications in the build path as class path
 * entries.
 * 
 * @since Version 1.2
 * @goal classpath
 * @phase compile
 */
public class ClassPathEntryAdditionMojo extends AbstractMojo {
    
    /**
     * Sets the directory for all the jsxm specifications. The default value is
     * the specdirectory under src/spec.
     * 
     * @parameter default-value="${basedir}/src/spec/"
     * @required
     */
    private String jsxmSpecRootDir;
    
    /**
     * The root directory of the project.
     * 
     * @parameter default-value="${basedir}"
     * @required
     */
    private String baseDir;
    
    /**
     * The way that specifications will be 
     * viewed in the package explorer of Eclipse IDE.
     * 
     * @parameter default-value="false"
     * @required
     */
    private boolean topView;
    
    /* (non-Javadoc)
     * @see org.apache.maven.plugin.Mojo#execute()
     */
    public void execute() throws MojoExecutionException, MojoFailureException {
        JSXMFacade.getInstance().addSpecificationToClassPath(baseDir, jsxmSpecRootDir,topView);
    }
    
}
