package org.jsxm.maven.plugin.mojo;

import org.apache.maven.plugin.AbstractMojo;
import org.apache.maven.plugin.MojoExecutionException;
import org.apache.maven.plugin.MojoFailureException;
import org.jsxm.maven.plugin.jsxmtool.JSXMFacade;

/**
 * @author Konstantinos Margaritis
 * 
 */

/**
 * Goal that generates the jsxm specification test and transforms them.
 *  External implementation support. Java implementation that is not provided
 *  in the src folder of the project will use the import of the Adapter as a package,
 *  and add the jsxm as root package. External implementation is processed 
 *  automatically in case local implementation is not found. In case the implementations
 *  can not be found by the IDE or is not available in the classpath you 
 *  should provide it in the classpath in order to be used by the Adapters. 
 * 
 * @since Version 1.0
 * @goal generate
 * @execute goal="compile"
 * @phase generate-test-resources
 */
public class GenerationMojo extends AbstractMojo {
       
    /**
     * 
     * 
     * @parameter default-value="false"
     * @readonly
     */
    @SuppressWarnings("unused")
    private boolean generateReport;
    
    /* (non-Javadoc)
     * @see org.apache.maven.plugin.Mojo#execute()
     */
    public void execute() throws MojoExecutionException, MojoFailureException {
        JSXMFacade.getInstance().generateTests();
        JSXMFacade.getInstance().transformTests();
    }
}
